5/**
6 * @file vehicle.c
7 * @brief Vehicle simulation implementation
8 *
9 * This file provides a complete vehicle simulation using ODE's
10 * Hinge2 joints to simulate wheels with suspension. The vehicle
11 * consists of a chassis body connected to wheel bodies via
12 * simulated suspension joints.
13 *
14 * @author Chris Camacho (codifies - http://bedroomcoders.co.uk/)
15 * @date 2021-2026
16 *
17 * @section vehicle_structure Vehicle Structure
18 *
19 * The vehicle is composed of:
20 * - 1 chassis body (box shape)
21 * - 4 wheel bodies (cylinder shape)
22 * - 1 anti-sway mass (lowered center of gravity)
23 * - 4 Hinge2 joints (suspension + steering)
24 * - 1 Fixed joint (anti-sway attachment)
25 *
26 * @section control Control
27 *
28 * Use UpdateVehicle() to control:
29 * - accel: Forward/backward acceleration
30 * - steer: Steering angle
31 */

34/**
35 * @brief Create a physics-based vehicle
36 *
37 * Creates a complete vehicle simulation with chassis, wheels,
38 * and suspension. The vehicle uses Hinge2 joints for realistic
39 * wheel suspension and steering mechanics.
40 *
41 * @param pctx Pointer to physics context
42 * @param ctx Pointer to graphics context
43 * @param pos Initial position of the vehicle
44 * @param carScale Dimensions of the chassis (x=length, y=height, z=width)
45 * @param wheelRadius Radius of the wheels
46 * @param wheelWidth Width of the wheels
47 * @return Pointer to newly created vehicle structure
48 *
49 * @note Front wheels are driving wheels
50 * @note Front wheels also provide steering
51 * @note Anti-sway mass lowers center of gravity to prevent tipping
52 *
53 * @see UpdateVehicle
54 * @see FreeVehicle
55 */
56vehicle* CreateVehicle(PhysicsContext* pctx, struct GraphicsContext* ctx, Vector3 pos, Vector3 carScale, float wheelRadius, float wheelWidth)
57{
58 // Extract ODE world and space from the framework context
59 dWorldID world = pctx->world;
60 dSpaceID space = pctx->space;
61
62 vehicle* car = RL_MALLOC(sizeof(vehicle));
63
64 car->bodyCount = 6;
65
66 // Get textures for vehicle parts
67 Texture* chassisTex = &ctx->boxTextures[0]; // crate.png
68 Texture* wheelTex = &ctx->cylinderTextures[1]; // cylinder2.png
69 Texture* markerTex = &ctx->boxTextures[1]; // grid.png
70
71 // Calculate wheel radius relative to car height if not explicitly specified
72 // Don't scale down wheels that are appropriately sized
73 float scaledWheelRadius = wheelRadius;
74
75 // Main Chassis
76 dMass m;
77 dMassSetBox(&m, 1, carScale.x, carScale.y, carScale.z);
78 dMassAdjust(&m, 150); // 150kg chassis
79
80 car->bodies[0] = dBodyCreate(world);
81 dBodySetMass(car->bodies[0], &m);
82 dBodySetAutoDisableFlag(car->bodies[0], 0);
83 dBodySetPosition(car->bodies[0], pos.x, pos.y, pos.z);
84
85 car->geoms[0] = dCreateBox(space, carScale.x, carScale.y, carScale.z);
86 dGeomSetBody(car->geoms[0], car->bodies[0]);
87 dGeomSetData(car->geoms[0], CreateGeomInfo(true, chassisTex, 1.0f, 1.0f));
88
89 // Front indicator (visual aid to tell front from back)
90 dGeomID frontGeom = dCreateBox(space, 0.2, 0.2, 0.2);
91 dGeomSetBody(frontGeom, car->bodies[0]);
92 dGeomSetOffsetPosition(frontGeom, carScale.x/2 - 0.1, carScale.y/2 + 0.1, 0);
93 dGeomSetData(frontGeom, CreateGeomInfo(true, markerTex, 1.0f, 1.0f));
94 car->geoms[6] = frontGeom;
95
96 // Anti-Sway / Low CoG Mass (Body 5)
97 // This mass lowers the center of gravity to prevent tipping
98 car->bodies[5] = dBodyCreate(world);
99 dMassSetSphere(&m, 1, 0.5);
100 dMassAdjust(&m, 100);
101 dBodySetMass(car->bodies[5], &m);
102 dBodySetPosition(car->bodies[5], pos.x, pos.y - 3.0, pos.z);
103 // no geom for this body it doesn't collide
104
105
106 // Fixed joint to attach balance mass to chassis
107 car->joints[4] = dJointCreateFixed(world, 0);
108 dJointAttach(car->joints[4], car->bodies[0], car->bodies[5]);
109 dJointSetFixed(car->joints[4]);
110
111 // Wheels (Bodies 1-4)
112 dMassSetCylinder(&m, 1, 3, scaledWheelRadius, wheelWidth);
113 dMassAdjust(&m, 2);
114
115 // Calculate wheel positions relative to car scale
116 // X is forward/back (length), Z is left/right (width)
117 float wheelYOffset = -0.5; // Fixed offset below chassis center - match original
118 float offsetX = carScale.x * 0.35; // Length-wise (front/rear) along X
119 float offsetZ = carScale.z * 0.7; // Width-wise (left/right) along Z - wider stance
120 float wheelOffsets[4][3] = {
121 { offsetX, wheelYOffset, -offsetZ }, // Front Left
122 { offsetX, wheelYOffset, offsetZ }, // Front Right
123 { -offsetX, wheelYOffset, -offsetZ }, // Rear Left
124 { -offsetX, wheelYOffset, offsetZ } // Rear Right
125 };
126
127 for(int i = 0; i < 4; i++)
128 {
129 int bIdx = i + 1;
130 car->bodies[bIdx] = dBodyCreate(world);
131 dBodySetMass(car->bodies[bIdx], &m);
132 dBodySetFiniteRotationMode(car->bodies[bIdx], 1);
133 dBodySetAutoDisableFlag(car->bodies[bIdx], 0);
134 dBodySetPosition(car->bodies[bIdx], pos.x + wheelOffsets[i][0], pos.y + wheelOffsets[i][1], pos.z + wheelOffsets[i][2]);
135
136 car->geoms[bIdx] = dCreateCylinder(space, scaledWheelRadius, wheelWidth);
137 dGeomSetBody(car->geoms[bIdx], car->bodies[bIdx]);
138 dGeomSetData(car->geoms[bIdx], CreateGeomInfo(true, wheelTex, 1.0f, 1.0f));
139 geomInfo* gi = dGeomGetData(car->geoms[bIdx]);
140 gi->surface = &gSurfaces[SURFACE_RUBBER];
141
142 // Hinge2 Joints (Joints 0-3) ---
143 car->joints[i] = dJointCreateHinge2(world, 0);
144 dJointAttach(car->joints[i], car->bodies[0], car->bodies[bIdx]);
145 dJointSetHinge2Anchor(car->joints[i], pos.x + wheelOffsets[i][0], pos.y + wheelOffsets[i][1], pos.z + wheelOffsets[i][2]);
146
147 dReal axis1[] = { 0, 1, 0 }; // Steering axis (Y - up/down)
148 dReal axis2[] = { 0 , 0, ((i % 2) == 0) ? -1 : 1 };
149 dJointSetHinge2Axes(car->joints[i], axis1, axis2);
150
151 // Soften suspension to help prevent the car from bouncing or flipping
152 dJointSetHinge2Param(car->joints[i], dParamSuspensionERP, 0.9);
153 dJointSetHinge2Param(car->joints[i], dParamSuspensionCFM, 0.002);
154
155 if (i < 2) { // Front wheel steering limits
156 dJointSetHinge2Param(car->joints[i], dParamLoStop, -.75);
157 dJointSetHinge2Param(car->joints[i], dParamHiStop, .75);
158 dJointSetHinge2Param(car->joints[i], dParamFMax, 15000);
159 } else { // Rear wheels - LOCK steering axis completely
160 dJointSetHinge2Param(car->joints[i], dParamLoStop, 0.0);
161 dJointSetHinge2Param(car->joints[i], dParamHiStop, 0.0); // Same as LoStop = locked
162 }
163
164 }
165
166 // Link bodies to entities so drawBodies and FreeVehicle function correctly
167 for (int i = 0; i < 6; i++) {
168 entity* ent = RL_MALLOC(sizeof(entity));
169 ent->body = car->bodies[i];
170 dBodySetData(car->bodies[i], ent);
171 ent->node = clistAddNode(pctx->objList, ent); //
172 }
173
174 return car;
175}

177/**
178 * @brief Update vehicle control inputs
179 *
180 * Applies acceleration and steering to the vehicle. This should
181 * be called every frame with current control inputs.
182 *
183 * @param car Pointer to the vehicle structure
184 * @param accel Acceleration input (-1.0 to 1.0, negative for reverse)
185 * @param steer Steering input (-1.0 to 1.0, negative for left)
186 *
187 * @note Set accel to 0 for coasting (wheels spin free)
188 * @note Front wheels provide steering and all-wheel drive
189 */
190void UpdateVehicle(vehicle *car, float accel, float steer)
191{
192 // Match original force values
193 float driveTorque = 800.0f;
194 float steeringForce = 800.0f;//500.0f;
195 float steerFactor = 10.0f;
196
197 // If no acceleration input, let the wheels spin free (coasting)
198 // Otherwise, it acts like a permanent brake
199 float currentTorque = (accel == 0) ? 0.0f : driveTorque;
200
201 // Front Wheels (0, 1) - Driving
202
203 dJointSetHinge2Param(car->joints[0], dParamVel2, -accel);
204 dJointSetHinge2Param(car->joints[1], dParamVel2, accel);
205 dJointSetHinge2Param(car->joints[0], dParamFMax2, currentTorque);
206 dJointSetHinge2Param(car->joints[1], dParamFMax2, currentTorque);
207
208 // Rear Wheels (2, 3) - Driving
209 dJointSetHinge2Param(car->joints[2], dParamVel2, -accel);
210 dJointSetHinge2Param(car->joints[3], dParamVel2, accel);
211 dJointSetHinge2Param(car->joints[2], dParamFMax2, currentTorque);
212 dJointSetHinge2Param(car->joints[3], dParamFMax2, currentTorque);
213
214
215 // Steering Logic
216 for(int i = 0; i < 2; i++) {
217 dReal v = steer - dJointGetHinge2Angle1(car->joints[i]);
218 v *= steerFactor;
219
220 dJointSetHinge2Param(car->joints[i], dParamFMax, steeringForce);
221 dJointSetHinge2Param(car->joints[i], dParamVel, v);
222 }
223
224
225}

233void FreeVehicle(PhysicsContext* pctx, vehicle* car)
234{
235 if (!car) return;
236
237 for (int i = 0; i < car->bodyCount; i++) {
238 // Get the entity wrapper attached to the ODE body
239 entity* ent = (entity*)dBodyGetData(car->bodies[i]);
240
241 if (ent) {
242 // Remove the node from the framework's render list
243 clistDeleteNode(pctx->objList, &ent->node);
244 FreeBodyAndGeoms(car->bodies[i]);
245 RL_FREE(ent);
246 }
247 }
248 RL_FREE(car);
249}

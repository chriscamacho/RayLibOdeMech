25// Create a light and get shader locations
26Light CreateLight(int type, Vector3 position, Vector3 target, Color color, Shader shader)
27{
28 Light light = { 0 };
29
30 if (lightsCount < MAX_LIGHTS)
31 {
32 light.enabled = true;
33 light.type = type;
34 light.position = position;
35 light.target = target;
36 light.color = color;
37
38 // NOTE: Lighting shader naming must be the provided ones
39 light.enabledLoc = GetShaderLocation(shader, TextFormat("lights[%i].enabled", lightsCount));
40 light.typeLoc = GetShaderLocation(shader, TextFormat("lights[%i].type", lightsCount));
41 light.positionLoc = GetShaderLocation(shader, TextFormat("lights[%i].position", lightsCount));
42 light.targetLoc = GetShaderLocation(shader, TextFormat("lights[%i].target", lightsCount));
43 light.colorLoc = GetShaderLocation(shader, TextFormat("lights[%i].color", lightsCount));
44
45 UpdateLightValues(shader, light);
46
47 lightsCount++;
48 }
49
50 return light;
51}

53// Send light properties to shader
54// NOTE: Light shader locations should be available
55void UpdateLightValues(Shader shader, Light light)
56{
57 // Send to shader light enabled state and type
58 SetShaderValue(shader, light.enabledLoc, &light.enabled, SHADER_UNIFORM_INT);
59 SetShaderValue(shader, light.typeLoc, &light.type, SHADER_UNIFORM_INT);
60
61 // Send to shader light position values
62 float position[3] = { light.position.x, light.position.y, light.position.z };
63 SetShaderValue(shader, light.positionLoc, position, SHADER_UNIFORM_VEC3);
64
65 // Send to shader light target position values
66 float target[3] = { light.target.x, light.target.y, light.target.z };
67 SetShaderValue(shader, light.targetLoc, target, SHADER_UNIFORM_VEC3);
68
69 // Send to shader light color values
70 float color[4] = { (float)light.color.r/(float)255, (float)light.color.g/(float)255,
71 (float)light.color.b/(float)255, (float)light.color.a/(float)255 };
72 SetShaderValue(shader, light.colorLoc, color, SHADER_UNIFORM_VEC4);
73}

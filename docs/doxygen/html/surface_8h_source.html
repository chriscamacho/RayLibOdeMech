23/**
24 * @file surface.h
25 * @brief Physics surface material definitions and lookup table.
26 * * Defines the physical properties for different world surfaces to be
27 * used in collision and friction calculations.
28 */
29
30#ifndef SURFACE_H
31#define SURFACE_H
32
33/**
34 * @enum SurfaceType
35 * @brief Unique identifiers for different physical materials.
36 * * These constants are used as indices into the gSurfaces array.
37 * @note SURFACE_COUNT must always remain the last element to allow for loop iteration.
38 */
39typedef enum
40{
41 SURFACE_WOOD = 0, /**< Hard organic surface with moderate friction. */
42 SURFACE_METAL, /**< Smooth metallic surface with high density. */
43 SURFACE_ICE, /**< Low-friction crystalline surface. */
44 SURFACE_RUBBER, /**< High-friction elastic surface. */
45 SURFACE_EARTH, /**< Granular, high-damping surface (soil/dirt). */
46 SURFACE_COUNT /**< The total number of defined surfaces. */
47} SurfaceType;
48
49/**
50 * @struct SurfaceMaterial
51 * @brief Physical coefficients defining how an object interacts with a surface.
52 * * This structure holds constants used by the physics engine to resolve
53 * collisions and sliding movement.
54 * * * @par Example Usage:
55 * @code
56 * entity* testBox = CreateBox(physCtx, graphics, (Vector3){1,1,1}, pos, Vector3Zero(), 20.f);
57 * dGeomID testGeom = dBodyGetFirstGeom(testBox->body);
58 * geomInfo* boxInfo = dGeomGetData(testGeom);
59 * boxInfo->surface = &gSurfaces[SURFACE_ICE];
60 * @endcode
61 * @note the surface should be set on every geom attached to a body you expect to
62 * collide, it defaults to SURFACE_EARTH and is not needed for things like triggers
63 * or geom's maked as not colliding
64 */
65typedef struct SurfaceMaterial
66{
67 float friction; /**< Coefficient of friction (mu). */
68 float bounce; /**< Coefficient of restitution. Defines energy loss on impact. */
69 float bounce_vel; /**< bounce velocity factor. */
70 float slip1; /**< Primary slip coefficient. */
71 float slip2; /**< Secondary slip coefficient. */
72} SurfaceMaterial;
73
74/**
75 * @brief Global lookup table for surface properties.
76 * This array is indexed by @ref SurfaceType. and contains default
77 * values for each surface type
78 */
79extern SurfaceMaterial gSurfaces[SURFACE_COUNT];

81#endif // SURFACE_H

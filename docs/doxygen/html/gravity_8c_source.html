41Vector3 GetSpawnOnSphereBorder(Vector3 center, float radius)
42{
43 Vector3 randDir = {
44 rndf(-100, 100),
45 rndf(-100, 100),
46 rndf(-100, 100)
47 };
48
49 randDir = Vector3Normalize(randDir);
50
51 float spawnDist = radius * 0.9f;
52 Vector3 offset = Vector3Scale(randDir, spawnDist);
53 return Vector3Add(center, offset);
54}

56entity* CreateOrbiter(PhysicsContext* physCtx, GraphicsContext* graphics)
57{
58 Vector3 p = GetSpawnOnSphereBorder(gravPoint, gravSize * 0.9f);
59 entity* e = CreateSphere(physCtx,graphics,.2, p, Vector3Zero(), 100.f);
60 geomInfo* gi = dGeomGetData(dBodyGetFirstGeom(e->body));
61 gi->surface = &gSurfaces[SURFACE_RUBBER];
62
63 Vector3 toSpawn = Vector3Subtract(p, gravPoint);
64 Vector3 fDir = Vector3Normalize(toSpawn);
65
66 // Create a random tangent (perpendicular to the radius)
67 Vector3 axis = { rndf(-1,1), rndf(-1,1), rndf(-1,1)};
68 if (fabsf(Vector3DotProduct(fDir, axis)) > 0.9f) axis = (Vector3){ 1, 0, 0 };
69
70 Vector3 tangent = Vector3Normalize(Vector3CrossProduct(fDir, axis));
71
72 // Calculate Orbital Speed: v = sqrt( (GravityAccel * Radius) )
73 // This gives a roughly stable circular orbit
74 float orbitSpeed = sqrtf(Vector3Length(G) * Vector3Length(toSpawn));
75
76 // Apply some(!) of the initial linear velocity to the ODE body
77 Vector3 velocity = Vector3Scale(tangent, orbitSpeed / 6.0f);
78 dBodySetLinearVel(e->body, velocity.x, velocity.y, velocity.z);
79
80 // using user data for point of the trails
81 e->data = MemAlloc(sizeof(Vector3) * trailSize);
82 Vector3* v = e->data;
83 for (int i = 0; i < trailSize; i++) {
84 v[i] = p;
85 }
86
87 return e;
88}

167void updateGravity(PhysicsContext* physCtx, GraphicsContext* graphics )
168{
169 StepPhysics(physCtx);
170
171 cnode_t* node = physCtx->objList->head;
172
173 while (node != NULL)
174 {
175 entity* ent = node->data;
176 dBodyID bdy = ent->body;
177 cnode_t* next = node->next; // get the next node now in case we delete this one
178 const dReal* pos = dBodyGetPosition(bdy);
179
180 Vector3* pv = (Vector3*)pos; // not sure if I like this!
181 Vector3 dist = Vector3Subtract(gravPoint, *pv);
182 float d = Vector3Length(dist);
183
184 if (d < gravSize) {
185 dBodySetGravityMode(bdy, 0);
186
187 Vector3 f = Vector3Normalize(dist);
188 Vector3Scale(f, Vector3Length(G));
189 dMass M;
190 dBodyGetMass(bdy, &M);
191 f = Vector3Scale(f, M.mass);
192 if (d < kullSize) {
193 // this is just to make it more interesting so they
194 // don't all clump in the centre it also helps
195 // to regenrate the object orbit by adding more energy
196 //f = Vector3Negate(f); // push away if too close
197 //f = Vector3Scale(f, 8.0f);
198
199 // as an alternatice you can use this with a smaller radius
200 free(ent->data);
201 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
202 CreateOrbiter(physCtx, graphics);
203 node = next;
204 continue;
205 }
206 dBodyAddForce(bdy, f.x, f.y, f.z);
207 } else {
208 dBodySetGravityMode(bdy, 1); // normal gravity
209 }
210
211 if(pos[1]<-10) {
212 free(ent->data);
213 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
214 CreateOrbiter(physCtx, graphics);
215 node = next;
216 continue;
217 }
218
219 if (frameCount % 4 == 0) {
220 Vector3* v = ent->data;
221 for (int i = 1; i < trailSize; i++) {
222 v[i-1] = v[i];
223 }
224 v[trailSize-1] = (Vector3){pos[0], pos[1], pos[2]};
225 }
226 node = next;
227 }
228
229}

27#define screenWidth 1920/1.2
28#define screenHeight 1080/1.2
29
30
31
32int main(void)
33{
34 // init
35 PhysicsContext* physCtx = CreatePhysics();
36 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE Sandbox");
37 SetupCamera(graphics);
38
39 // Create ground plane
40 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
41 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
42 dGeomSetData(planeGeom, CreateGeomInfo(true, &graphics->groundTexture, 25.0f, 25.0f));
43 clistAddNode(physCtx->statics, planeGeom);
44
45 // Create random simple objects with random textures
46 for (int i = 0; i < NUM_OBJ; i++) {
47 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
48 }
49
50
51 // manually create a 3 section piston
52 // TLDR skip down to the multi piston creation !
53 entity* box1 = CreateBox(physCtx, graphics,(Vector3){4,1,1}, (Vector3){0,1.6,0}, (Vector3){0,0,0}, 2);
54 entity* box2 = CreateBox(physCtx, graphics,(Vector3){4,.9,.9}, (Vector3){.1,1.7,.2}, (Vector3){0,0,0}, 2);
55 entity* box3 = CreateBox(physCtx, graphics,(Vector3){4,.8,.8}, (Vector3){.2,1.8,.4}, (Vector3){0,0,0}, 2);
56
57 // the direction each box is offset - the axis of the piston
58 Vector3 dir = (Vector3){.1,.1,.2};
59 dir = Vector3Normalize(dir);
60
61 SetBodyOrientation(box1->body, dir);
62 SetBodyOrientation(box2->body, dir);
63 SetBodyOrientation(box3->body, dir);
64 // anchor box1 to the world
65 PinEntityToWorld(physCtx, box1);
66
67
68 dJointID piston1 = CreatePiston(physCtx, box1, box2,1000);
69 SetPistonLimits(piston1, 0, 3.7);
70 dJointID piston2 = CreatePiston(physCtx, box2, box3,1000);
71 SetPistonLimits(piston2, 0, 3.7);
72
73
74 // because box 1,2 and 3 intersect we must filter out their collisions
75 // normally with joints the two attached bodies don't collide, however
76 // this doesn't help with box 1 vs box 3 for example
77 dGeomID g1 = dBodyGetFirstGeom(box1->body);
78 dGeomID g2 = dBodyGetFirstGeom(box2->body);
79 dGeomID g3 = dBodyGetFirstGeom(box3->body);
80
81 dGeomSetCategoryBits(g1, PISTON_GROUP);
82 dGeomSetCollideBits(g1, WORLD_GROUP);
83
84 dGeomSetCategoryBits(g2, PISTON_GROUP);
85 dGeomSetCollideBits(g2, WORLD_GROUP);
86
87 dGeomSetCategoryBits(g3, PISTON_GROUP);
88 dGeomSetCollideBits(g3, WORLD_GROUP);
89
90
91
92 // create a multi piston and pin it to the world
93 MultiPiston* mp = CreateMultiPiston(physCtx, graphics,
94 (Vector3){4,1.4,0}, // position
95 (Vector3){.5,.5,0}, // direction (or axis) of the piston
96 6, // number of sections
97 0.5, // distance each section travels
98 2, // base width
99 1000); // stength
100
101 PinEntityToWorld(physCtx, mp->sections[0]);
102
103
104
105 while (!WindowShouldClose())
106 {
107 dJointSetSliderParam(piston1, dParamVel, 0.0);
108 dJointSetSliderParam(piston2, dParamVel, 0.0);
109 SetMultiPistonVelocity(mp, 0);
110
111 float pSpeed = 1;
112 if (IsKeyDown(KEY_LEFT_ALT)) pSpeed = 4;
113
114 if (IsKeyDown(KEY_I)) {
115 dJointSetSliderParam(piston1, dParamVel, -pSpeed);
116 dJointSetSliderParam(piston2, dParamVel, -pSpeed);
117 dBodyEnable(box2->body);
118 SetMultiPistonVelocity(mp, -pSpeed/16.0);
119 }
120 if (IsKeyDown(KEY_O)) {
121 dJointSetSliderParam(piston1, dParamVel, pSpeed);
122 dJointSetSliderParam(piston2, dParamVel, pSpeed);
123 dBodyEnable(box2->body);
124 SetMultiPistonVelocity(mp, pSpeed/16.0);
125 }
126
127 bool spcdn = IsKeyDown(KEY_SPACE);
128 cnode_t* node = physCtx->objList->head;
129 while (node != NULL) {
130 entity* ent = node->data;
131 dBodyID bdy = ent->body;
132 cnode_t* next = node->next; // get the next node now in case we delete this one
133 const dReal* pos = dBodyGetPosition(bdy);
134
135 if (spcdn) {
136 // apply force if the space key is held
137 const dReal* v = dBodyGetLinearVel(bdy);
138 if (v[1] < 10 && pos[1]<10) { // cap upwards velocity and don't let it get too high
139 dBodyEnable (bdy); // case its gone to sleep
140 dMass mass;
141 dBodyGetMass (bdy, &mass);
142 // give some object more force than others
143 float f = rndf(8,20) * mass.mass;
144 dBodyAddForce(bdy, rndf(-f,f), f*10, rndf(-f,f));
145 }
146 }
147
148 if(pos[1]<-10) {
149 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
150 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
151 }
152
153 node = next;
154 }
155
156 UpdateCameraControl(graphics);
157 StepPhysics(physCtx);
158
159 // drawing
160 BeginDrawing();
161 ClearBackground(BLACK);
162 BeginMode3D(graphics->camera);
163 DrawBodies(graphics, physCtx);
164 DrawStatics(graphics, physCtx);
165 EndMode3D();
166
167 DrawText("Press O and P to move the piston", 10, 40, 20, RAYWHITE);
168
169 EndDrawing();
170 }
171
172
173 FreeMultiPiston(mp);
174 FreePhysics(physCtx);
175 FreeGraphics(graphics);
176 CloseWindow();
177 return 0;
178}

31int main(void)
32{
33 // init
34 PhysicsContext* physCtx = CreatePhysics();
35 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE Sandbox");
36 SetupCamera(graphics);
37
38 // Create ground plane
39 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
40 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
41 dGeomSetData(planeGeom, CreateGeomInfo(true, &graphics->groundTexture, 25.0f, 25.0f));
42 clistAddNode(physCtx->statics, planeGeom);
43
44
45
46 while (!WindowShouldClose())
47 {
48 UpdateCameraControl(graphics);
49 StepPhysics(physCtx);
50
51 // drawing
52 BeginDrawing();
53 ClearBackground(BLACK);
54 BeginMode3D(graphics->camera);
55 DrawBodies(graphics, physCtx);
56 DrawStatics(graphics, physCtx);
57 EndMode3D();
58
59 DrawText("This does nothing use it for your own experiments!", 10, 40, 20, RAYWHITE);
60
61 EndDrawing();
62 }
63
64 FreePhysics(physCtx);
65 FreeGraphics(graphics);
66 CloseWindow();
67 return 0;
68}

24#include <stdbool.h>
25#include <math.h>

29#define screenWidth 1920/1.2
30#define screenHeight 1080/1.2
31
32// 0x01 = world 0x02 = pistons
33
34#define WALL_GROUP 0x00000004
35#define ROTOR_GROUP 0x00000008
36
37int main(void)
38{
39 // Initialization
40 //--------------------------------------------------------------------------------------
41
42 PhysicsContext* physCtx = CreatePhysics();
43 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE Sandbox");
44 SetupCamera(graphics);
45
46 // set up for the items in the world
47 //--------------------------------------------------------------------------------------
48
49 // Create ground "plane"
50 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE+2, PLANE_THICKNESS, PLANE_SIZE+2);
51 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
52 geomInfo* groundInfo = CreateGeomInfo(true, &graphics->groundTexture, 50.0f, 50.0f);
53 groundInfo->surface = &gSurfaces[SURFACE_EARTH];
54 dGeomSetData(planeGeom, groundInfo);
55 dGeomSetCategoryBits(planeGeom, WALL_GROUP);
56 dGeomSetCollideBits(planeGeom, WORLD_GROUP); // Ignore WALL_ROTOR (and WALL!)
57
58 clistAddNode(physCtx->statics, planeGeom);
59
60 // Create random simple objects with random textures
61 for (int i = 0; i < NUM_OBJ; i++) {
62 entity* e = CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL & ~SHAPE_DUMBBELL);
63 geomInfo* gi = dGeomGetData(dBodyGetFirstGeom(e->body));
64 gi->surface = &gSurfaces[SURFACE_RUBBER];
65 }
66
67
68 entity* rotor = CreateBox(physCtx, graphics,
69 (Vector3){7, 6,1}, // size
70 (Vector3){0,0,0}, // pos
71 (Vector3){0,0,0}, // rot
72 4); // mass
73
74 // offset the geom so it is being rotated from one end
75 dGeomID rgeom = dBodyGetFirstGeom(rotor->body);
76 dGeomSetOffsetPosition(rgeom, 3.5, 0, 0);
77 geomInfo* gi = dGeomGetData(rgeom);
78 gi->surface = &gSurfaces[SURFACE_ICE];
79 dGeomSetCategoryBits(rgeom, ROTOR_GROUP);
80 dGeomSetCollideBits(rgeom, WORLD_GROUP); // Ignore WALL_GROUP (and rotors too)
81
82 dJointID joint_hinge = CreateRotor(physCtx, rotor, 0, (Vector3){0,1,0});
83
84 float physTime = 0;
85
86 //--------------------------------------------------------------------------------------
87 //
88 // Main game loop
89 //
90 //--------------------------------------------------------------------------------------
91 float rotorSpeed = -1;
92 bool wiper = false;
93 while (!WindowShouldClose()) // Detect window close button or ESC key
94 {
95 //--------------------------------------------------------------------------------------
96 // Update
97 //----------------------------------------------------------------------------------
98
99 // baked in controls (example camera)
100 UpdateCameraControl(graphics);
101 if (IsKeyPressed(KEY_O)) {
102 wiper = !wiper;
103 }
104
105 if (wiper) {
106 float ha = dJointGetHingeAngle (joint_hinge);
107 if (ha < -M_PI_2/2) rotorSpeed = fabs(rotorSpeed);
108 if (ha > M_PI_2/2) rotorSpeed = -fabs(rotorSpeed);
109 }
110
111 if (IsKeyPressed(KEY_R)) {
112 rotorSpeed = -rotorSpeed;
113 }
114
115 if (IsKeyDown(KEY_P)) {
116 // will get slowed if pushing too much...
117 dJointSetHingeParam(joint_hinge, dParamVel, rotorSpeed*4.0f);
118 } else {
119 dJointSetHingeParam(joint_hinge, dParamVel, rotorSpeed);
120 }
121
122 bool spcdn = IsKeyDown(KEY_SPACE); // cache space key status (don't look up for each object iterration
123 cnode_t* node = physCtx->objList->head;
124
125 while (node != NULL) {
126 entity* ent = node->data;
127 dBodyID bdy = ent->body;
128 cnode_t* next = node->next; // get the next node now in case we delete this one
129 const dReal* pos = dBodyGetPosition(bdy);
130 if (spcdn) {
131 // apply force if the space key is held
132 const dReal* v = dBodyGetLinearVel(bdy);
133 if (v[1] < 10 && pos[1]<10) { // cap upwards velocity and don't let it get too high
134 dBodyEnable (bdy); // case its gone to sleep
135 dMass mass;
136 dBodyGetMass (bdy, &mass);
137 // give some object more force than others
138 float f = rndf(8,20) * mass.mass;
139 dBodyAddForce(bdy, rndf(-f,f), f*10, rndf(-f,f));
140 }
141 }
142
143
144 if(pos[1]<-10) {
145 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
146 entity* e = CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL & ~SHAPE_DUMBBELL);
147 geomInfo* gi = dGeomGetData(dBodyGetFirstGeom(e->body));
148 gi->surface = &gSurfaces[SURFACE_RUBBER];
149 }
150
151 node = next;
152 }
153
154 // Step the physics
155 //----------------------------------------------------------------------------------
156
157 physTime = GetTime();
158 int pSteps = StepPhysics(physCtx);
159 physTime = GetTime() - physTime;
160
161
162 // Draw
163 //----------------------------------------------------------------------------------
164
165 BeginDrawing();
166
167 ClearBackground(BLACK);
168
169 BeginMode3D(graphics->camera);
170 DrawBodies(graphics, physCtx);
171 DrawStatics(graphics, physCtx);
172 EndMode3D();
173
174
175 if (pSteps > maxPsteps) DrawText("WARNING CPU overloaded lagging real time", 10, 0, 20, RED);
176 DrawText(TextFormat("%2i FPS", GetFPS()), 10, 20, 20, WHITE);
177 DrawText("Press SPACE to apply force to objects", 10, 60, 20, WHITE);
178
179 DrawText(TextFormat("Phys steps per frame %i",pSteps), 10, 120, 20, WHITE);
180 DrawText(TextFormat("Phys time per frame %f",physTime), 10, 140, 20, WHITE);
181 DrawText(TextFormat("total time per frame %f",GetFrameTime()), 10, 160, 20, WHITE);
182
183 EndDrawing();
184
185 }
186
187
188 // De-Initialization
189 //--------------------------------------------------------------------------------------
190 FreePhysics(physCtx);
191 FreeGraphics(graphics);
192
193 CloseWindow(); // Close window and OpenGL context
194
195 return 0;
196}

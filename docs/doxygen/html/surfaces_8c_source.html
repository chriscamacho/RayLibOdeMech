27#define screenWidth 1920/1.2
28#define screenHeight 1080/1.2
29

32int main(void)
33{
34 // Initialization
35 //--------------------------------------------------------------------------------------
36
37 // Physics context, holds all physics state
38 PhysicsContext* physCtx = CreatePhysics();
39 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE");
40
41 SetupCamera(graphics);
42
43 graphics->camera.position.y = graphics->camera.position.y - 8.f;
44
45 // set up for the items in the world
46 //--------------------------------------------------------------------------------------
47
48 // Create ground "plane"
49 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
50 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
51 dMatrix3 R_plane;
52 dRFromAxisAndAngle(R_plane, 1, 0, 0, M_PI * 0.125);
53 dGeomSetRotation(planeGeom, R_plane);
54 geomInfo* groundInfo = CreateGeomInfo(true, &graphics->groundTexture, 25.0f, 25.0f);
55 groundInfo->surface = &gSurfaces[SURFACE_EARTH];
56 dGeomSetData(planeGeom, groundInfo);
57
58
59 clistAddNode(physCtx->statics, planeGeom);
60
61
62 for (SurfaceType i = 0; i < SURFACE_COUNT; i++) {
63 Vector3 pos = (Vector3){i*2, 2, 0};
64 entity* testBox = CreateBox(physCtx, graphics, (Vector3){1,1,1}, pos, Vector3Zero(), 20.f);
65 SetEntitySurfaces(testBox, &gSurfaces[i]);
66 }
67
68
69 float physTime = 0;
70
71 //--------------------------------------------------------------------------------------
72 //
73 // Main game loop
74 //
75 //--------------------------------------------------------------------------------------
76 while (!WindowShouldClose()) // Detect window close button or ESC key
77 {
78 //--------------------------------------------------------------------------------------
79 // Update
80 //----------------------------------------------------------------------------------
81
82 // baked in controls (example only camera!)
83 UpdateExampleCamera(graphics);
84
85 bool spcdn = IsKeyDown(KEY_SPACE); // cache space key status (don't look up for each object iterration
86 cnode_t* node = physCtx->objList->head;
87
88 while (node != NULL) {
89 entity* ent = node->data;
90 dBodyID bdy = ent->body;
91
92 SetEntityHew(ent, WHITE);
93
94 cnode_t* next = node->next; // get the next node now in case we delete this one
95 const dReal* pos = dBodyGetPosition(bdy);
96 if (spcdn) {
97 // apply force if the space key is held
98 const dReal* v = dBodyGetLinearVel(bdy);
99 if (v[1] < 10 && pos[1]<10) { // cap upwards velocity and don't let it get too high
100 dBodyEnable (bdy); // case its gone to sleep
101 dMass mass;
102 dBodyGetMass (bdy, &mass);
103 // give some object more force than others
104 float f = rndf(8,20) * mass.mass;
105 dBodyAddForce(bdy, rndf(-f,f), f*10, rndf(-f,f));
106 }
107 }
108
109
110 if(pos[1]<-10) {
111 // would be more efficient to just reuse the object and
112 // reposition it with zeroed velocities
113 // but this is used to aid testing
114 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
115 //CreateRandomEntity(physCtx, graphics, (Vector3){rndf(5, 11), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
116 }
117
118 node = next;
119 }
120
121 // Step the physics
122 //----------------------------------------------------------------------------------
123
124 physTime = GetTime();
125 int pSteps = StepPhysics(physCtx);
126 physTime = GetTime() - physTime;
127
128
129 // Draw
130 //----------------------------------------------------------------------------------
131
132 BeginDrawing();
133
134 ClearBackground(BLACK);
135
136 BeginMode3D(graphics->camera);
137 DrawBodies(graphics, physCtx);
138 DrawStatics(graphics, physCtx);
139
140 EndMode3D();
141
142
143
144 if (pSteps > maxPsteps) DrawText("WARNING CPU overloaded lagging real time", 10, 0, 20, RED);
145 DrawText(TextFormat("%2i FPS", GetFPS()), 10, 20, 20, WHITE);
146 DrawText("Press SPACE to apply force to objects", 10, 60, 20, WHITE);
147 DrawText("Wood, Metal, Ice, Rubber, Earth on Earth", 10, 80, 20, WHITE);
148 DrawText(TextFormat("Phys steps per frame %i",pSteps), 10, 120, 20, WHITE);
149 DrawText(TextFormat("Phys time per frame %f",physTime), 10, 140, 20, WHITE);
150 DrawText(TextFormat("total time per frame %f",GetFrameTime()), 10, 160, 20, WHITE);
151
152 EndDrawing();
153
154 }
155
156
157 // De-Initialization
158 //--------------------------------------------------------------------------------------
159 FreePhysics(physCtx);
160 FreeGraphics(graphics);
161
162 CloseWindow(); // Close window and OpenGL context
163
164 return 0;
165}

132int main(void)
133{
134 // init
135 PhysicsContext* physCtx = CreatePhysics();
136 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE Sandbox");
137 SetupCamera(graphics);
138 graphics->camera.position = (Vector3){0.f,5.f,8.f};
139 SetCameraYaw(M_PI/4);
140
141 const Vector3 dropPoint = (Vector3){0.f,12.f,0.f};
142
143 LoadLevel(physCtx,graphics);
144
145
146
147 MultiPiston** lift1 = CreateLift(physCtx,
148 graphics,
149 (Vector3){0.f, 0.4f, 0.0f},
150 0,
151 1000.0f);
152
153
154 MultiPiston** lift2 = CreateLift(physCtx,
155 graphics,
156 (Vector3){6.f, 4.f, 8.f},
157 PI,
158 1000.0f);
159
160 MultiPiston** lift3 = CreateLift(physCtx,
161 graphics,
162 (Vector3){-10.f, 11.f, -18.f},
163 0,
164 1000.0f);
165
166 int frameCount = 0;
167 int released = 0;
168 const int maxRelesed = 16;
169 const float ballSize = .6f;
170
171 while (!WindowShouldClose())
172 {
173 for(int i=0; i < MAX_PISTON; i++) {
174 float offset = 0.f;
175 // every other piston is 180 degrees out of phase
176 if (i % 2 == 0) offset = M_PI;
177 SetMultiPistonVelocity(lift1[i], sin(((float)frameCount)/60.0f+offset));
178 SetMultiPistonVelocity(lift2[i], sin(((float)frameCount)/60.0f+offset));
179 SetMultiPistonVelocity(lift3[i], sin(((float)frameCount)/60.0f+offset));
180 }
181
182
183 if (released < maxRelesed) {
184 if (frameCount % 420 == 0) {
185 entity* e = CreateSphere(physCtx, graphics, ballSize, dropPoint, Vector3Zero(), 1);
186 dBodySetAutoDisableFlag(e->body, 0);
187 geomInfo* gi = dGeomGetData(dBodyGetFirstGeom(e->body));
188 gi->surface = &gSurfaces[SURFACE_METAL];
189 released++;
190 }
191 }
192 frameCount++;
193
194 UpdateCameraControl(graphics);
195 StepPhysics(physCtx);
196
197 cnode_t* node = physCtx->objList->head;
198
199 while (node != NULL) {
200 entity* ent = node->data;
201 dBodyID bdy = ent->body;
202 cnode_t* next = node->next; // get the next node now in case we delete this one
203 const dReal* pos = dBodyGetPosition(bdy);
204
205 if(pos[1]<-10) {
206 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
207 //entity* e = CreateSphere(physCtx, graphics, 0.55, dropPoint, Vector3Zero(), 1);
208 //dBodySetAutoDisableFlag(e->body, 0);
209 //geomInfo* gi = dGeomGetData(dBodyGetFirstGeom(e->body));
210 //gi->surface = &gSurfaces[SURFACE_RUBBER];
211 released--;
212 }
213
214 node = next;
215 }
216
217 // drawing
218 BeginDrawing();
219 ClearBackground(BLACK);
220 BeginMode3D(graphics->camera);
221 DrawBodies(graphics, physCtx);
222 DrawStatics(graphics, physCtx);
223 EndMode3D();
224
225 DrawText("Marbles!", 10, 40, 20, RAYWHITE);
226
227 EndDrawing();
228 }
229
230 FreePhysics(physCtx);
231 FreeGraphics(graphics);
232
233 // TODO make this easier !
234 for (int i = 0; i < MAX_PISTON; i++)
235 {
236 FreeMultiPiston(lift1[i]);
237 FreeMultiPiston(lift2[i]);
238 FreeMultiPiston(lift3[i]);
239 }
240 free(lift1);
241 free(lift2);
242 free(lift3);
243
244 CloseWindow();
245 return 0;
246}

249void LoadLevel(PhysicsContext* physCtx, GraphicsContext* graphics) {
250 int numElements = sizeof(levelData) / sizeof(LevelElement);
251
252 const float layerHeight = 1.17;
253
254 for (int i = 0; i < numElements; i++) {
255 LevelElement e = levelData[i];
256 float ea = e.angle + M_PI_2;
257
258 switch(e.type)
259 {
260 case SLOPE:
261 (void)e; // workaround for c99 decl after label
262 float pAngle = 30.0f * (M_PI / 180.0f);
263
264 if (ea > M_PI * 2.0f) ea -= M_PI * 2.0f;
265 dQuaternion q,q2,q3;
266 dQFromAxisAndAngle (q3, 1, 0, 0, pAngle);
267 dQFromAxisAndAngle (q2, 0, 1, 0, ea);
268 dQMultiply0 (q, q2, q3);
269 dMatrix4 R;
270 dQtoR(q, R);
271
272 float offX = 0.5f * cosf(ea);
273 float offZ = 0.5f * sinf(ea);
274
275 // Rail 1
276 Vector3 pos = (Vector3) {
277 (float)e.gridX * 2.0f + offX,
278 (float)e.gridY * layerHeight -.2f,
279 (float)e.gridZ * 2.0f - offZ};
280 dGeomID dg = CreateCylinderGeom(physCtx, graphics, 0.1f, 2.4f, pos);
281 geomInfo* gi = dGeomGetData(dg);
282 gi->surface = &gSurfaces[levelSurface];
283 dGeomSetRotation(dg, R);
284
285 // Rail 2
286 pos.x = (float)e.gridX * 2.0f - offX;
287 pos.z = (float)e.gridZ * 2.0f + offZ;
288 dGeomID dg2 = CreateCylinderGeom(physCtx, graphics, 0.1f, 2.4f, pos);
289 gi = dGeomGetData(dg2);
290 gi->surface = &gSurfaces[levelSurface];
291 dGeomSetRotation(dg2, R);
292
293 clistAddNode(physCtx->statics, dg);
294 clistAddNode(physCtx->statics, dg2);
295 break;
296
297 case STRAIGHT:
298
299 offX = 0.5f * cosf(ea);
300 offZ = 0.5f * sinf(ea);
301
302 pos = (Vector3) {
303 (float)e.gridX * 2.0f + offX,
304 (float)e.gridY * layerHeight +0.4f,
305 (float)e.gridZ * 2.0f - offZ};
306
307 dg = CreateCylinderGeom(physCtx, graphics, 0.1f, 2, pos);
308 gi = dGeomGetData(dg);
309 gi->surface = &gSurfaces[levelSurface];
310
311 pos.x = (float)e.gridX * 2.0f - offX;
312 pos.z = (float)e.gridZ * 2.0f + offZ;
313
314 dg2 = CreateCylinderGeom(physCtx, graphics, 0.1f, 2, pos);
315 gi = dGeomGetData(dg2);
316 gi->surface = &gSurfaces[levelSurface];
317
318 // always rotate as Z axis cylinders need extra half pi rotation
319 dQFromAxisAndAngle(q, 0, 1, 0, e.angle+M_PI_2);
320 dGeomSetQuaternion(dg, q);
321 dGeomSetQuaternion(dg2, q);
322
323 clistAddNode(physCtx->statics, dg);
324 clistAddNode(physCtx->statics, dg2);
325 break;
326
327 case CORNER:
328 // this was just a brain melt !
329 offX = (float)e.gridX * 2.0f;
330 offZ = (float)e.gridZ * 2.0f;
331
332 // Stick with the rot + M_PI_2 that gave us consistency
333 float rot = e.angle + M_PI_2;
334 float s = sinf(rot);
335 float c = cosf(rot);
336
337 for (int j = 0; j < 3; j++) {
338 float pAngle = (j * (M_PI_2 / 3.0f)) + (M_PI_2 / 6.0f);
339
340 float lx1 = -1.0f + 0.5f * cosf(pAngle);
341 float lz1 = -1.0f + 0.5f * sinf(pAngle);
342 float lx2 = -1.0f + 1.6f * cosf(pAngle);
343 float lz2 = -1.0f + 1.6f * sinf(pAngle);
344
345 pos.x = offX - (lx1 * c + lz1 * s);
346 pos.y = (float)e.gridY * layerHeight + 0.4f;
347 pos.z = offZ - (lz1 * c - lx1 * s);
348
349 dQFromAxisAndAngle(q, 0, 1, 0, rot - pAngle);
350 dQtoR(q, R);
351
352 // Inner Rail
353 dg = CreateCylinderGeom(physCtx, graphics, 0.1f, 0.261f, pos);
354 gi = dGeomGetData(dg);
355 gi->surface = &gSurfaces[levelSurface];
356 dGeomSetRotation(dg, R);
357 clistAddNode(physCtx->statics, dg);
358
359 pos.x = offX - (lx2 * c + lz2 * s);
360 pos.y += 0.5f;
361 pos.z = offZ - (lz2 * c - lx2 * s);
362
363 // Outer Rail
364 dg2 = CreateCylinderGeom(physCtx, graphics, 0.1f, 1.f, pos);
365 gi = dGeomGetData(dg2);
366 gi->surface = &gSurfaces[levelSurface];
367 dGeomSetRotation(dg2, R);
368 clistAddNode(physCtx->statics, dg2);
369 }
370 break;
371 }
372
373 }
374}

378MultiPiston** CreateLift(PhysicsContext* physCtx,
379 GraphicsContext* graphics,
380 Vector3 position,
381 float yaw,
382 float strength)
383{
384 Vector3 upAxis = (Vector3){0.f, 1.f, 0.f};
385
386 // Allocate multipiston array using raylib allocator
387 MultiPiston** mp = (MultiPiston**)MemAlloc(sizeof(MultiPiston*) * MAX_PISTON);
388
389 dMatrix3 R_yaw;
390 dRFromAxisAndAngle(R_yaw, 0, 1, 0, yaw);
391
392 // Pistons
393 for (int i = 0; i < MAX_PISTON; i++)
394 {
395 Vector3 localPos = (Vector3){10.f, -3.8f + i*1.4f, 7.6f + i*1.6f};
396
397 Vector3 worldPos = Vector3Add(
398 position,
399 Vector3RotateByAxisAngle(localPos, upAxis, yaw)
400 );
401
402 Vector3 axis = Vector3RotateByAxisAngle(
403 (Vector3){1.f,0.f,0.f},
404 upAxis,
405 yaw
406 );
407
408 mp[i] = CreateMultiPiston(physCtx, graphics,
409 worldPos,
410 axis,
411 3,
412 2.0f,
413 1.2f,
414 strength);
415
416 dMatrix3 R_local;
417 dMatrix3 R_final;
418
419 dRFromEulerAngles(R_local,
420 0,
421 M_PI_2,
422 (-M_PI/16.f) * 7.f);
423
424 dMultiply0(R_final, R_yaw, R_local, 3, 3, 3);
425 dBodySetRotation(mp[i]->sections[0]->body, R_final);
426 PinEntityToWorld(physCtx, mp[i]->sections[0]);
427 }
428
429 // -------------------------------------------------
430 // Lift End
431 // -------------------------------------------------
432 {
433 Vector3 localPos = (Vector3){10.f, 5.f, 16.7f};
434
435 Vector3 worldPos = Vector3Add(
436 position,
437 Vector3RotateByAxisAngle(localPos, upAxis, yaw)
438 );
439
440 dGeomID liftEnd = CreateBoxGeom(
441 physCtx,
442 graphics,
443 (Vector3){2.f,.1f,4.f},
444 worldPos
445 );
446
447 dMatrix3 R_tilt;
448 dMatrix3 R_final;
449
450 dRFromAxisAndAngle(R_tilt, 1, 0, 0, -(M_PI/16.f) * 7.f);
451 dMultiply0(R_final, R_yaw, R_tilt, 3, 3, 3);
452 dGeomSetRotation(liftEnd, R_final);
453 clistAddNode(physCtx->statics, liftEnd);
454 }
455
456 // -------------------------------------------------
457 // Rail 1
458 // -------------------------------------------------
459 {
460 Vector3 localPos = (Vector3){9.2f, 2.f, 12.f};
461 Vector3 worldPos = Vector3Add(
462 position,
463 Vector3RotateByAxisAngle(localPos, upAxis, yaw)
464 );
465
466 dGeomID liftRail1 = CreateBoxGeom(
467 physCtx,
468 graphics,
469 (Vector3){.1f,.8f,13.f},
470 worldPos
471 );
472
473 dMatrix3 R_tilt;
474 dMatrix3 R_final;
475
476 dRFromAxisAndAngle(R_tilt, 1, 0, 0, -(M_PI/16.f) * 3.5f);
477 dMultiply0(R_final, R_yaw, R_tilt, 3, 3, 3);
478 dGeomSetRotation(liftRail1, R_final);
479 clistAddNode(physCtx->statics, liftRail1);
480 }
481
482 // -------------------------------------------------
483 // Rail 2
484 // -------------------------------------------------
485 {
486 Vector3 localPos = (Vector3){10.8f, 2.f, 12.f};
487
488 Vector3 worldPos = Vector3Add(
489 position,
490 Vector3RotateByAxisAngle(localPos, upAxis, yaw)
491 );
492
493 dGeomID liftRail2 = CreateBoxGeom(
494 physCtx,
495 graphics,
496 (Vector3){.1f,.8f,13.f},
497 worldPos
498 );
499
500 dMatrix3 R_tilt;
501 dMatrix3 R_final;
502
503 dRFromAxisAndAngle(R_tilt, 1, 0, 0, -(M_PI/16.f) * 3.5f);
504
505 dMultiply0(R_final, R_yaw, R_tilt, 3, 3, 3);
506
507 dGeomSetRotation(liftRail2, R_final);
508
509 clistAddNode(physCtx->statics, liftRail2);
510 }
511
512 return mp;

24/**
25 * @file ragdoll.c
26 * @brief Ragdoll physics simulation implementation
27 *
28 * This file provides a complete ragdoll (humanoid) physics simulation
29 * using ODE's joint system. The ragdoll consists of multiple body parts
30 * connected by hinge and universal joints, allowing for realistic
31 * human-like movement and posing.
32 *
33 * @author Chris Camacho (codifies - http://bedroomcoders.co.uk/)
34 * @date 2026
35 *
36 * @section ragdoll_structure Ragdoll Structure
37 *
38 * Body parts (10 total):
39 * - Head (sphere)
40 * - Torso (box)
41 * - Left/Right Upper Arm (capsule)
42 * - Left/Right Lower Arm (capsule)
43 * - Left/Right Upper Leg (capsule)
44 * - Left/Right Lower Leg (capsule)
45 *
46 * Joints (9 total):
47 * - Neck (hinge)
48 * - Left/Right Shoulder (universal)
49 * - Left/Right Elbow (hinge)
50 * - Left/Right Hip (universal)
51 * - Left/Right Knee (hinge)
52 *
53 * @section motor_control Motor Control
54 *
55 * The ragdoll supports motor control on all joints, making it
56 * suitable for neural network or reinforcement learning control.
57 * Use UpdateRagdollMotors() to apply forces to joints.
58 */
59

63// Get a spawn position within the defined ragdoll spawn volume
64
65/**
66 * @brief Get a random spawn position for ragdolls
67 *
68 * Generates a random position within the configured spawn volume
69 * for ragdoll objects. This helps spread out multiple ragdolls
70 * when they are created.
71 *
72 * @return Random Vector3 position within spawn bounds
73 *
74 * @note Spawn bounds are defined by RAGDOLL_SPAWN_* constants
75 * @see RAGDOLL_SPAWN_CENTER_X
76 * @see RAGDOLL_SPAWN_CENTER_Z
77 * @see RAGDOLL_SPAWN_HALF_EXTENT
78 * @see RAGDOLL_SPAWN_MIN_Y
79 * @see RAGDOLL_SPAWN_MAX_Y
80 */
81Vector3 GetRagdollSpawnPosition(void)
82{
83 Vector3 pos;
84 pos.x = rndf(RAGDOLL_SPAWN_CENTER_X - RAGDOLL_SPAWN_HALF_EXTENT, RAGDOLL_SPAWN_CENTER_X + RAGDOLL_SPAWN_HALF_EXTENT);
85 pos.y = rndf(RAGDOLL_SPAWN_MIN_Y, RAGDOLL_SPAWN_MAX_Y);
86 pos.z = rndf(RAGDOLL_SPAWN_CENTER_Z - RAGDOLL_SPAWN_HALF_EXTENT, RAGDOLL_SPAWN_CENTER_Z + RAGDOLL_SPAWN_HALF_EXTENT);
87 return pos;
88}

95/**
96 * @brief Create a ragdoll physics body
97 *
98 * Creates a complete humanoid ragdoll with articulated joints.
99 * The ragdoll is fully physics-simulated and can be used for
100 * character physics, death animations, or as a basis for
101 * reinforcement learning control.
102 *
103 * @param pctx Pointer to physics context
104 * @param ctx Pointer to graphics context
105 * @param position Initial position for the ragdoll
106 * @return Pointer to newly created RagDoll structure
107 *
108 * @note Bodies: 10 (head, torso, 2 upper arms, 2 lower arms, 2 upper legs, 2 lower legs)
109 * @note Joints: 9 (neck, shoulders, elbows, hips, knees)
110 * @note All bodies are registered with framework for rendering
111 *
112 * @see RagDoll
113 * @see UpdateRagdollMotors
114 * @see DrawRagdoll
115 * @see FreeRagdoll
116 */

122 ragdoll->motorCount = 0; // No motors initially, can be added for neural network control

124 // Allocate arrays for bodies, geoms, joints, and motors
125 ragdoll->bodies = RL_MALLOC(ragdoll->bodyCount * sizeof(dBodyID));

397// Update rag doll motors
398// motorForces array should have one value per joint for control
399
400/**
401 * @brief Update ragdoll joint motors
402 *
403 * Applies motor forces to the ragdoll's joints for control.
404 * This is useful for neural network or reinforcement learning
405 * control of the ragdoll.
406 *
407 * @param ragdoll Pointer to the ragdoll structure
408 * @param motorForces Array of motor force values (one per joint)
409 *
410 * @note Array should have ragdoll->jointCount elements for hinge joints
411 * @note For universal joints, uses ragdoll->jointCount * 2 elements
412 * @note Force of 0 disables the motor for that joint
413 *
414 * @see CreateRagdoll
415 */
416void UpdateRagdollMotors(RagDoll *ragdoll, float *motorForces)
417{
418 if (!ragdoll || !motorForces) return;
419
420 for (int i = 0; i < ragdoll->jointCount; i++) {
421 dJointID joint = ragdoll->joints[i];
422 int jointType = dJointGetType(joint);
423
424 // For hinge joints, set velocity and max force for motor control
425 if (jointType == dJointTypeHinge) {
426 dJointSetHingeParam(joint, dParamVel, motorForces[i]);
427 dJointSetHingeParam(joint, dParamFMax, fabsf(motorForces[i]) > 0.001f ? 50.0f : 0.0f);
428 }
429 // For universal joints, control both axes
430 else if (jointType == dJointTypeUniversal) {
431 // motorForces[i] controls axis 1, motorForces[i + jointCount] would control axis 2
432 dJointSetUniversalParam(joint, dParamVel, motorForces[i]);
433 dJointSetUniversalParam(joint, dParamVel2, motorForces[i + ragdoll->jointCount]);
434 dJointSetUniversalParam(joint, dParamFMax, fabsf(motorForces[i]) > 0.001f ? 50.0f : 0.0f);
435 dJointSetUniversalParam(joint, dParamFMax2, fabsf(motorForces[i + ragdoll->jointCount]) > 0.001f ? 50.0f : 0.0f);
436 }
437 }
438}

466// Free rag doll resources - properly destroys ODE objects before freeing wrapper arrays
467
468/**
469 * @brief Free a ragdoll and all its resources
470 *
471 * Properly destroys all ODE bodies, geoms, joints, and motors
472 * associated with the ragdoll, and frees the wrapper structure.
473 *
474 * @param ctx Pointer to physics context
475 * @param ragdoll Pointer to the ragdoll structure to free
476 *
477 * @note Removes ragdoll from framework's object list
478 * @note Frees all allocated memory for bodies, geoms, joints
479 *
480 * @see CreateRagdoll
481 */
482void FreeRagdoll(PhysicsContext* ctx, RagDoll *ragdoll)
483{
484 if (!ragdoll) return;
485
486 // Destroy ODE joints
487 if (ragdoll->joints) {
488 for (int i = 0; i < ragdoll->jointCount; i++) {
489 if (ragdoll->joints[i]) {
490 dJointDestroy(ragdoll->joints[i]);
491 }
492 }
493 }
494
495 // Destroy ODE motors
496 if (ragdoll->motors) {
497 for (int i = 0; i < ragdoll->motorCount; i++) {
498 if (ragdoll->motors[i]) {
499 dJointDestroy(ragdoll->motors[i]);
500 }
501 }
502 }
503
504 // Destroy ODE bodies and geoms (remove from framework tracking too)
505 for (int i=0; i<ragdoll->bodyCount; i++)
506 {
507 entity* ent = dBodyGetData(ragdoll->bodies[i]);
508 FreeBodyAndGeoms(ragdoll->bodies[i]);
509 clistDeleteNode(ctx->objList, &ent->node);
510 RL_FREE(ent);
511 }
512
513 // Free wrapper arrays
514 if (ragdoll->bodies) RL_FREE(ragdoll->bodies);
515 if (ragdoll->geoms) RL_FREE(ragdoll->geoms);
516 if (ragdoll->joints) RL_FREE(ragdoll->joints);
517 if (ragdoll->motors) RL_FREE(ragdoll->motors);
518
519
520
521 RL_FREE(ragdoll);
522}

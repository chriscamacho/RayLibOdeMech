28#define screenWidth 1920/1.2
29#define screenHeight 1080/1.2
30
31#define NRAGDOLLS 16
32
33
34int main(void)
35{
36 // Initialization
37 //--------------------------------------------------------------------------------------
38
39 PhysicsContext* physCtx = CreatePhysics();
40 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE Sandbox");
41 SetupCamera(graphics);
42
43
44
45 // set up for the items in the world
46 //--------------------------------------------------------------------------------------
47
48 // Create ground "plane"
49 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
50 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
51 dMatrix3 R_plane;
52 //dRFromAxisAndAngle(R_plane, 1, 0, -1, M_PI * 0.125);
53 dRFromAxisAndAngle(R_plane, 0, 0, 1, 0);
54 dGeomSetRotation(planeGeom, R_plane);
55 dGeomSetData(planeGeom, CreateGeomInfo(true, &graphics->groundTexture, 25.0f, 25.0f));
56
57 clistAddNode(physCtx->statics, planeGeom);
58
59 RagDoll* rd[NRAGDOLLS];
60 for (int i=0; i<NRAGDOLLS; i++) {
61 rd[i] = CreateRagdoll(physCtx, graphics, GetRagdollSpawnPosition());
62 }
63
64 float physTime = 0;
65
66 //--------------------------------------------------------------------------------------
67 //
68 // Main game loop
69 //
70 //--------------------------------------------------------------------------------------
71 while (!WindowShouldClose()) // Detect window close button or ESC key
72 {
73 //--------------------------------------------------------------------------------------
74 // Update
75 //----------------------------------------------------------------------------------
76
77 // baked in controls (example only camera!)
78 UpdateExampleCamera(graphics);
79
80
81 bool spcdn = IsKeyDown(KEY_SPACE); // cache space key status (don't look up for each object iterration
82
83 // Apply lifting force to ragdolls when space is held
84 if (spcdn) {
85 for (int i = 0; i < NRAGDOLLS; i++) {
86 if (rd[i] && rd[i]->bodies[RAGDOLL_HEAD]) {
87 dBodyEnable(rd[i]->bodies[RAGDOLL_HEAD]);
88 // Calculate total mass of all body parts in the ragdoll
89 float totalMass = 0.0f;
90 for (int j = 0; j < rd[i]->bodyCount; j++) {
91 if (rd[i]->bodies[j]) {
92 dMass partMass;
93 dBodyGetMass(rd[i]->bodies[j], &partMass);
94 totalMass += partMass.mass;
95 }
96 }
97 // Lift force based on total ragdoll mass (60 * total mass)
98 float liftForce = 60.0f * totalMass;
99 dBodyAddForce(rd[i]->bodies[RAGDOLL_HEAD],
100 rndf(-10, 10), liftForce + rndf(-5, 5), rndf(-10, 10));
101 }
102 }
103 }
104
105 // Reset rag dolls if they fall off the plane
106 for (int i = 0; i < NRAGDOLLS; i++) {
107 if (rd[i] && rd[i]->bodies[RAGDOLL_TORSO]) {
108 const dReal* pos = dBodyGetPosition(rd[i]->bodies[RAGDOLL_TORSO]);
109 if (pos[1] < -10) {
110 // Re-create rag doll at a new random spawn position
111 FreeRagdoll(physCtx, rd[i]); // remove framework resources
112 rd[i] = CreateRagdoll(physCtx, graphics, GetRagdollSpawnPosition());
113 }
114 }
115 }
116
117
118
119 // Step the physics
120 //----------------------------------------------------------------------------------
121
122 physTime = GetTime();
123 int pSteps = StepPhysics(physCtx);
124 physTime = GetTime() - physTime;
125
126
127 // Draw
128 //----------------------------------------------------------------------------------
129
130 BeginDrawing();
131
132 ClearBackground(BLACK);
133
134 BeginMode3D(graphics->camera);
135 DrawBodies(graphics, physCtx);
136 DrawStatics(graphics, physCtx);
137 EndMode3D();
138
139
140 if (pSteps > maxPsteps) DrawText("WARNING CPU overloaded lagging real time", 10, 0, 20, RED);
141 DrawText(TextFormat("%2i FPS", GetFPS()), 10, 20, 20, WHITE);
142 DrawText("Press SPACE to apply force to objects", 10, 60, 20, WHITE);
143
144 DrawText(TextFormat("Phys steps per frame %i",pSteps), 10, 120, 20, WHITE);
145 DrawText(TextFormat("Phys time per frame %f",physTime), 10, 140, 20, WHITE);
146 DrawText(TextFormat("total time per frame %f",GetFrameTime()), 10, 160, 20, WHITE);
147
148 EndDrawing();
149
150 }
151
152 // De-Initialization
153 //--------------------------------------------------------------------------------------
154 for (int i=0; i<NRAGDOLLS; i++) {
155 FreeRagdoll(physCtx,rd[i]);
156 }
157 FreePhysics(physCtx);
158 FreeGraphics(graphics);
159
160 CloseWindow(); // Close window and OpenGL context
161
162 return 0;
163}

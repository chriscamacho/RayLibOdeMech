27#define screenWidth 1920/1.2
28#define screenHeight 1080/1.2
29

32int main(void)
33{
34 // Initialization
35 //--------------------------------------------------------------------------------------
36
37 // Physics context, holds all physics state
38 PhysicsContext* physCtx = CreatePhysics();
39 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE");
40
41 SetupCamera(graphics);
42
43 graphics->camera.position.y = graphics->camera.position.y - 8.f;
44
45 // set up for the items in the world
46 //--------------------------------------------------------------------------------------
47
48 // Create ground "plane"
49
50 Texture dotTex = LoadTexture("data/dot.png");
51
52 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
53 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
54 SetGeomOrientationEuler(planeGeom, -22.5f, 0, 0);
55
56 geomInfo* groundInfo = CreateGeomInfo(true, &dotTex, 25.0f, 25.0f);
57 groundInfo->surface = &gSurfaces[SURFACE_ICE];
58 dGeomSetData(planeGeom, groundInfo);
59
60 clistAddNode(physCtx->statics, planeGeom);
61
62
63 RayCast* redCast = CreateRayCast(12, (Vector3){-5,.5,0},(Vector3){1,0,0}, 3);
64 // pre calculate end point as we're not moving the ray around
65 // (used for drawing the visual)
66 Vector3 redEnd = Vector3Scale(redCast->direction, redCast->length);
67 redEnd = Vector3Add(redEnd, redCast->position);
68
69 RayCast* greenCast = CreateRayCast(12, (Vector3){-2,.5,0},(Vector3){1,0,0}, 3);
70 Vector3 greenEnd = Vector3Scale(greenCast->direction, greenCast->length);
71 greenEnd = Vector3Add(greenEnd, greenCast->position);
72
73 RayCast* blueCast = CreateRayCast(12, (Vector3){-2,-0.5,2},(Vector3){1,0,0}, 7);
74 Vector3 blueEnd = Vector3Scale(blueCast->direction, blueCast->length);
75 blueEnd = Vector3Add(blueEnd, blueCast->position);
76
77 float physTime = 0;
78 int frameCount = 0;
79
80 //--------------------------------------------------------------------------------------
81 //
82 // Main game loop
83 //
84 //--------------------------------------------------------------------------------------
85 while (!WindowShouldClose()) // Detect window close button or ESC key
86 {
87 //--------------------------------------------------------------------------------------
88 // Update
89 //----------------------------------------------------------------------------------
90
91 // baked in controls (example camera)
92 UpdateCameraControl(graphics);
93
94 cnode_t* node = physCtx->objList->head;
95
96
97
98 while (node != NULL) {
99 entity* ent = node->data;
100 dBodyID bdy = ent->body;
101
102 cnode_t* next = node->next; // get the next node now in case we delete this one
103 const dReal* pos = dBodyGetPosition(bdy);
104
105 if(pos[1]<-10) {
106 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
107 }
108
109 node = next;
110 }
111
112 if (frameCount % 60 == 0) {
113 for (int i=0; i < 3; i++) {
114 Vector3 pos = (Vector3){-4.0f + i *4.0f, 6.5, -7.5};
115 entity* box = CreateBox(physCtx, graphics, (Vector3){1,1,1}, pos, Vector3Zero(), 20.f);
116 SetEntitySurfaces(box, &gSurfaces[SURFACE_ICE]);
117 }
118 }
119
120 // Step the physics
121 //----------------------------------------------------------------------------------
122
123 physTime = GetTime();
124 int pSteps = StepPhysics(physCtx);
125 physTime = GetTime() - physTime;
126
127 CastRay(physCtx, redCast);
128 for (int i = 0; i < redCast->count; i++) {
129 geomInfo* gi = dGeomGetData(redCast->hits[i].geom);
130 if (gi) gi->hew = RED;
131 }
132 CastRay(physCtx, greenCast);
133 for (int i = 0; i < greenCast->count; i++) {
134 geomInfo* gi = dGeomGetData(greenCast->hits[i].geom);
135 if (gi) gi->hew = GREEN;
136 }
137
138 CastRay(physCtx, blueCast);
139 for (int i = 0; i < blueCast->count; i++) {
140 geomInfo* gi = dGeomGetData(blueCast->hits[i].geom);
141 if (gi) gi->hew = BLUE;
142 }
143 // Draw
144 //----------------------------------------------------------------------------------
145
146 BeginDrawing();
147
148 ClearBackground(BLACK);
149
150 BeginMode3D(graphics->camera);
151 DrawBodies(graphics, physCtx);
152 DrawStatics(graphics, physCtx);
153
154 // rays probably more useful invisible but in this example we show them
155 DrawLine3D(redCast->position, redEnd, RED);
156 DrawLine3D(greenCast->position, greenEnd, GREEN);
157 DrawLine3D(blueCast->position, blueEnd, BLUE);
158 EndMode3D();
159
160
161 if (pSteps > maxPsteps) DrawText("WARNING CPU overloaded lagging real time", 10, 0, 20, RED);
162 DrawText(TextFormat("%2i FPS", GetFPS()), 10, 20, 20, WHITE);
163 DrawText(TextFormat("Phys steps per frame %i",pSteps), 10, 120, 20, WHITE);
164 DrawText(TextFormat("Phys time per frame %f",physTime), 10, 140, 20, WHITE);
165 DrawText(TextFormat("total time per frame %f",GetFrameTime()), 10, 160, 20, WHITE);
166
167 EndDrawing();
168
169 frameCount++;
170 }
171
172 UnloadTexture(dotTex);
173 free(redCast);
174 free(greenCast);
175 free(blueCast);
176 // De-Initialization
177 //--------------------------------------------------------------------------------------
178 FreePhysics(physCtx);
179 FreeGraphics(graphics);
180
181 CloseWindow(); // Close window and OpenGL context
182
183 return 0;
184}

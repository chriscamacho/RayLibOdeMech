27#define screenWidth 1920/1.2
28#define screenHeight 1080/1.2
29

32int main(void)
33{
34 PhysicsContext* physCtx = CreatePhysics();
35 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE Sandbox");
36 SetupCamera(graphics);
37
38
39 // Create ground plane
40 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
41 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
42 dGeomSetData(planeGeom, CreateGeomInfo(true, &graphics->groundTexture, 25.0f, 25.0f));
43 clistAddNode(physCtx->statics, planeGeom);
44
45 // Initial random objects
46 for (int i = 0; i < NUM_OBJ; i++) {
47 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
48 }
49
50 while (!WindowShouldClose())
51 {
52 UpdateCameraControl(graphics);
53
54 // placing new objects
55 Vector3 forward = Vector3Normalize(Vector3Subtract(graphics->camera.target, graphics->camera.position));
56 Vector3 spawnPos = Vector3Add(graphics->camera.position, Vector3Scale(forward, 3.0f));
57 Vector3 defaultRot = { 0, GetCameraYaw(), 0 };
58
59 if (IsKeyPressed(KEY_ONE)) CreateBox(physCtx, graphics, (Vector3){0.5, 0.5, 0.5}, spawnPos, defaultRot, 10.0f);
60 if (IsKeyPressed(KEY_TWO)) CreateSphere(physCtx, graphics, 0.4f, spawnPos, defaultRot, 10.0f);
61 if (IsKeyPressed(KEY_THREE)) CreateCylinder(physCtx, graphics, 0.3f, 1.0f, spawnPos, defaultRot, 10.0f);
62 if (IsKeyPressed(KEY_FOUR)) CreateCapsule(physCtx, graphics, 0.3f, 1.0f, spawnPos, defaultRot, 10.0f);
63 if (IsKeyPressed(KEY_FIVE)) CreateDumbbell(physCtx, graphics, 0.1f, 1.0f, 0.3f, spawnPos, defaultRot, 10.0f);
64
65 // picking / pushing
66 Vector3 hitPoint = { 0 };
67 entity* picked = PickEntity(physCtx, graphics, &hitPoint);
68 bool hasHit = (picked != NULL);
69
70 if (IsMouseButtonDown(MOUSE_LEFT_BUTTON) && hasHit && picked->body != NULL) {
71 dBodyEnable(picked->body);
72 Vector3 dir = Vector3Normalize(Vector3Subtract(hitPoint, graphics->camera.position));
73 Vector3 force = Vector3Scale(dir, 500.0f);
74 dBodyAddForceAtPos(picked->body, force.x, force.y, force.z, hitPoint.x, hitPoint.y, hitPoint.z);
75 }
76
77 bool spcdn = IsKeyDown(KEY_SPACE); // cache space key status (don't look up for each object iterration
78
79 // check world bounds and add force if needed
80 cnode_t* node = physCtx->objList->head;
81 while (node != NULL) {
82 entity* ent = node->data;
83 dBodyID bdy = ent->body;
84 cnode_t* next = node->next; // get the next node now in case we delete this one
85 const dReal* pos = dBodyGetPosition(bdy);
86 if (spcdn) {
87 // apply force if the space key is held
88 const dReal* v = dBodyGetLinearVel(bdy);
89 if (v[1] < 10 && pos[1]<10) { // cap upwards velocity and don't let it get too high
90 dBodyEnable (bdy); // case its gone to sleep
91 dMass mass;
92 dBodyGetMass (bdy, &mass);
93 // give some object more force than others
94 float f = rndf(8,20) * mass.mass;
95 dBodyAddForce(bdy, rndf(-f,f), f*10, rndf(-f,f));
96 }
97 }
98
99
100 if(pos[1] < -10) {
101 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
102 }
103 node = next;
104 }
105
106 StepPhysics(physCtx);
107
108 // drawing
109 BeginDrawing();
110 ClearBackground(BLACK);
111 BeginMode3D(graphics->camera);
112 DrawBodies(graphics, physCtx);
113 DrawStatics(graphics, physCtx);
114
115 // Target indicator (Red dot)
116 if (hasHit) {
117 DrawSphere(hitPoint, 0.05f, RED);
118 } else {
119 // Spawn point preview (Grey dot)
120 DrawSphereEx(spawnPos, 0.05f, 8, 8, DARKGRAY);
121 }
122 EndMode3D();
123
124 DrawText("1-5: Spawn Objects | LMB: Push | Space: Apply Force", 10, 40, 20, RAYWHITE);
125
126 EndDrawing();
127 }
128
129 FreePhysics(physCtx);
130 FreeGraphics(graphics);
131 CloseWindow();
132 return 0;
133}

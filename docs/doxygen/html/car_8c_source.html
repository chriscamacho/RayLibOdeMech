27#define screenWidth 1920/1.2
28#define screenHeight 1080/1.2

33int main(void)
34{
35 // Initialization
36 //--------------------------------------------------------------------------------------
37
38 // Physics context, holds all physics state
39 PhysicsContext* physCtx = CreatePhysics();
40 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE");
41 SetupCamera(graphics);
42
43 // set up for the items in the world
44 //--------------------------------------------------------------------------------------
45
46 // the graphical mesh for the ground must be unloaded by the user
47 Model ground = LoadModel("data/ground2.obj");
48
49 // framework looks after the physics stuff and rendering
50 CreateStaticTrimesh(physCtx, graphics, ground, &graphics->groundTexture, 2.5f);
51
52
53 // vehicle* CreateVehicle(PhysicsContext* pctx, struct GraphicsContext* ctx, Vector3 pos, Vector3 carScale, float wheelRadius, float wheelWidth);
54 vehicle* car = CreateVehicle(physCtx, graphics, (Vector3){15, 6, 15},(Vector3){3.5, 0.5, 2.4}, .8, .6);
55
56
57 // Create random simple objects with random textures
58 for (int i = 0; i < NUM_OBJ; i++) {
59 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
60 }
61
62
63 float physTime = 0;
64
65 //--------------------------------------------------------------------------------------
66 //
67 // Main game loop
68 //
69 //--------------------------------------------------------------------------------------
70 while (!WindowShouldClose()) // Detect window close button or ESC key
71 {
72 //--------------------------------------------------------------------------------------
73 // Update
74 //----------------------------------------------------------------------------------
75
76 // baked in controls (example only camera!)
77 UpdateVehicleCamera(graphics, car);
78
79 bool spcdn = IsKeyDown(KEY_SPACE); // cache space key status (don't look up for each object iterration
80 cnode_t* node = physCtx->objList->head;
81
82 while (node != NULL) {
83 entity* ent = node->data;
84 dBodyID bdy = ent->body;
85 cnode_t* next = node->next; // get the next node now in case we delete this one
86 const dReal* pos = dBodyGetPosition(bdy);
87 if (spcdn) {
88 // apply force if the space key is held
89 const dReal* v = dBodyGetLinearVel(bdy);
90 if (v[1] < 10 && pos[1]<10) { // cap upwards velocity and don't let it get too high
91 dBodyEnable (bdy); // case its gone to sleep
92 dMass mass;
93 dBodyGetMass (bdy, &mass);
94 // give some object more force than others
95 float f = rndf(80,200) * mass.mass;
96 dBodyAddForce(bdy, rndf(-f,f), f*10, rndf(-f,f));
97 }
98 }
99
100
101 if(pos[1]<-10) {
102 // would be more efficient to just reuse the object and
103 // reposition it with zeroed velocities
104 // but this is used to aid testing
105 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
106 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
107 }
108
109 node = next;
110 }
111
112 float accel = 0;
113 if (IsKeyDown(KEY_DOWN)) accel = -100.0f;
114 if (IsKeyDown(KEY_UP)) accel = 100.0f;
115 float steer = 0;
116 if (IsKeyDown(KEY_LEFT)) steer = -0.4f;
117 if (IsKeyDown(KEY_RIGHT)) steer = 0.4f;
118
119 UpdateVehicle(car, accel, steer);
120
121 // Step the physics
122 //----------------------------------------------------------------------------------
123
124 physTime = GetTime();
125 int pSteps = StepPhysics(physCtx);
126 physTime = GetTime() - physTime;
127
128
129 // Draw
130 //----------------------------------------------------------------------------------
131
132 BeginDrawing();
133
134 ClearBackground(BLACK);
135
136 BeginMode3D(graphics->camera);
137 DrawBodies(graphics, physCtx);
138 DrawStatics(graphics, physCtx);
139 EndMode3D();
140
141
142 if (pSteps > maxPsteps) DrawText("WARNING CPU overloaded lagging real time", 10, 0, 20, RED);
143 DrawText(TextFormat("%2i FPS", GetFPS()), 10, 20, 20, WHITE);
144 DrawText("Press SPACE to apply force to objects", 10, 60, 20, WHITE);
145
146 DrawText(TextFormat("Phys steps per frame %i",pSteps), 10, 120, 20, WHITE);
147 DrawText(TextFormat("Phys time per frame %f",physTime), 10, 140, 20, WHITE);
148 DrawText(TextFormat("total time per frame %f",GetFrameTime()), 10, 160, 20, WHITE);
149
150 EndDrawing();
151
152 }
153
154
155 // De-Initialization
156 //--------------------------------------------------------------------------------------
157
158 FreeVehicle(physCtx, car);
159 UnloadModel(ground);
160
161 FreePhysics(physCtx);
162 FreeGraphics(graphics);
163
164 CloseWindow(); // Close window and OpenGL context
165
166 return 0;
167}

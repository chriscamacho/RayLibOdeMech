28#define screenWidth 1920/1.2
29#define screenHeight 1080/1.2
30
31

33int main(void)
34{
35 // init
36 PhysicsContext* physCtx = CreatePhysics();
37 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE Sandbox");
38 SetupCamera(graphics);
39
40 // Create ground plane
41 dGeomID planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
42 dGeomSetPosition(planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
43 dGeomSetData(planeGeom, CreateGeomInfo(true, &graphics->groundTexture, 25.0f, 25.0f));
44 clistAddNode(physCtx->statics, planeGeom);
45
46 // Create random simple objects with random textures
47 //for (int i = 0; i < NUM_OBJ; i++) {
48 // CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
49 //}
50
51
52 entity* rotor = CreateCylinder(physCtx, graphics, 2.f, 1.f, (Vector3){0,.6,0}, (Vector3){M_PI_2,0,0}, 100);
53 dJointID rotorJoint = CreateRotor(physCtx, rotor, 0, (Vector3){0,1,0});
54 dBodySetAutoDisableFlag(rotor->body, 0);
55
56 entity* elivate = CreateSphere(physCtx, graphics, 1.f, (Vector3){0,1.6,0}, (Vector3){M_PI_2,0,0}, 1);
57 dJointID elivateJoint = CreateRotor(physCtx, elivate, rotor, (Vector3){1,0,0});
58 dBodySetAutoDisableFlag(elivate->body, 0);
59
60
61 // create a multi piston and pin it to the world
62 // extends from .22 to 5.54 with the furthest extent being plus section height / 2
63 // bottom for first section is section[0] position minus section height / 2
64 MultiPiston* upperArm = CreateMultiPiston(physCtx, graphics,
65 (Vector3){0,3.5,0}, // position
66 (Vector3){0,1,0}, // direction (or axis) of the piston
67 4, // number of sections
68 2, // distance each section travels
69 1, // base width
70 1000); // stength
71 dBodySetAutoDisableFlag(upperArm->sections[0]->body, 0);
72
73 PinEntities(physCtx, elivate, upperArm->sections[0]);
74
75 entity* elbow = CreateSphere(physCtx, graphics, 1.f, (Vector3){0,5.8,0}, (Vector3){M_PI_2,0,0}, 1);
76 dJointID elbowJoint = CreateRotor(physCtx, elbow, upperArm->sections[upperArm->count-1], (Vector3){1,0,0});
77 dBodySetAutoDisableFlag(elbow->body, 0);
78
79
80 MultiPiston* foreArm = CreateMultiPiston(physCtx, graphics,
81 (Vector3){0,7.8,0}, // position
82 (Vector3){0,1,0}, // direction (or axis) of the piston
83 4, // number of sections
84 2, // distance each section travels
85 1, // base width
86 1000); // stength
87 dBodySetAutoDisableFlag(foreArm->sections[0]->body, 0);
88
89 PinEntities(physCtx, elbow, foreArm->sections[0]);
90
91 RotorPID elivatePID = CreateRotorPID(100.0f, 1.6f, 3.f, -M_PI_2 + 0.2f, M_PI_2 - 0.2f);
92 RotorPID elbowPID = CreateRotorPID(100.0f, 1.6f,3.f, -M_PI_2 + 0.2f, M_PI_2 - 0.2f);
93
94 entity* grabber = CreateSphere(physCtx, graphics, 1.f, (Vector3){0,10.2f,0}, (Vector3){M_PI_2,0,0}, 1);
95 PinEntities(physCtx, grabber, foreArm->sections[foreArm->count-1]);
96
97 while (!WindowShouldClose())
98 {
99
100 dJointSetHingeParam(rotorJoint, dParamVel, 0);
101 if (IsKeyDown(KEY_I)) dJointSetHingeParam(rotorJoint, dParamVel, 1.0);
102 if (IsKeyDown(KEY_O)) dJointSetHingeParam(rotorJoint, dParamVel, -1.0);
103
104 if (IsKeyDown(KEY_K)) elivatePID.targetAngle += 0.01;
105 if (IsKeyDown(KEY_L)) elivatePID.targetAngle -= 0.01;
106 UpdateRotorPID(&elivatePID, elivateJoint);
107
108 SetMultiPistonVelocity(upperArm, 0);
109 if (IsKeyDown(KEY_COMMA)) SetMultiPistonVelocity(upperArm, 1);
110 if (IsKeyDown(KEY_PERIOD)) SetMultiPistonVelocity(upperArm, -1);
111
112 if (IsKeyDown(KEY_Y)) elbowPID.targetAngle += 0.01;
113 if (IsKeyDown(KEY_U)) elbowPID.targetAngle -= 0.01;
114 UpdateRotorPID(&elbowPID, elbowJoint);
115
116 SetMultiPistonVelocity(foreArm, 0);
117 if (IsKeyDown(KEY_H)) SetMultiPistonVelocity(foreArm, 1);
118 if (IsKeyDown(KEY_J)) SetMultiPistonVelocity(foreArm, -1);
119
120 bool spcdn = IsKeyDown(KEY_SPACE);
121 cnode_t* node = physCtx->objList->head;
122 while (node != NULL) {
123 entity* ent = node->data;
124 dBodyID bdy = ent->body;
125 cnode_t* next = node->next; // get the next node now in case we delete this one
126 const dReal* pos = dBodyGetPosition(bdy);
127
128 if (spcdn) {
129 // apply force if the space key is held
130 const dReal* v = dBodyGetLinearVel(bdy);
131 if (v[1] < 10 && pos[1]<10) { // cap upwards velocity and don't let it get too high
132 dBodyEnable (bdy); // case its gone to sleep
133 dMass mass;
134 dBodyGetMass (bdy, &mass);
135 // give some object more force than others
136 float f = rndf(8,20) * mass.mass;
137 dBodyAddForce(bdy, rndf(-f,f), f*10, rndf(-f,f));
138 }
139 }
140
141 if(pos[1]<-10) {
142 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
143 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
144 }
145
146 node = next;
147 }
148
149 UpdateCameraControl(graphics);
150 StepPhysics(physCtx);
151
152 // drawing
153 BeginDrawing();
154 ClearBackground(BLACK);
155 BeginMode3D(graphics->camera);
156 DrawBodies(graphics, physCtx);
157 DrawStatics(graphics, physCtx);
158 EndMode3D();
159
160 //DrawText("Press O and P to move the piston", 10, 40, 20, RAYWHITE);
161 float ea = dJointGetHingeAngle(elivateJoint);
162 DrawText(TextFormat("Elivate angle %f",ea), 10, 120, 20, WHITE);
163 ea = dJointGetHingeAngle(rotorJoint);
164 DrawText(TextFormat("Rotor angle %f",ea), 10, 140, 20, WHITE);
165 ea = dJointGetHingeAngle(elbowJoint);
166 DrawText(TextFormat("elbow angle %f",ea), 10, 160, 20, WHITE);
167
168 {
169 const dReal* op1 = dBodyGetPosition(foreArm->sections[0]->body);
170 const dReal* op2 = dBodyGetPosition(foreArm->sections[foreArm->count-1]->body);
171 Vector3 dv = (Vector3){op1[0]-op2[0], op1[1]-op2[1], op1[2]-op2[2]};
172 float l = Vector3Length(dv);
173 DrawText(TextFormat("forarm length %f", l), 10, 180, 20, WHITE);
174 }
175
176 {
177 const dReal* op1 = dBodyGetPosition(upperArm->sections[0]->body);
178 const dReal* op2 = dBodyGetPosition(upperArm->sections[upperArm->count-1]->body);
179 Vector3 dv = (Vector3){op1[0]-op2[0], op1[1]-op2[1], op1[2]-op2[2]};
180 float l = Vector3Length(dv);
181 DrawText(TextFormat("upperArm length %f", l), 10, 200, 20, WHITE);
182 }
183
184 {
185 const dReal* op1 = dBodyGetPosition(grabber->body);
186 DrawText(TextFormat("grabber position %f, %f, %f", op1[0], op1[1], op1[2]), 10, 220, 20, WHITE);
187 }
188
189 EndDrawing();
190 }
191
192
193 FreeMultiPiston(upperArm);
194 FreePhysics(physCtx);
195 FreeGraphics(graphics);
196 CloseWindow();
197 return 0;
198}

24/**
25 * @file collision.c
26 * @brief Collision detection and response for the physics framework
27 *
28 * This file handles all collision detection between physics bodies
29 * using ODE's collision system. It provides the nearCallback function
30 * that is called by ODE during collision detection.
31 *
32 * @author Chris Camacho (codifies - http://bedroomcoders.co.uk/)
33 * @date 2026
34 *
35 * @note This module uses dSpaceCollide for broad-phase collision detection
36 * @note Maximum contacts per collision pair is limited to 8
37 */
38
39#include "raylibODE.h"
40
41#define MAX_CONTACTS 8
42
43void nearCallback(void *data, dGeomID o1, dGeomID o2)
44{
45
46 dBodyID b1 = dGeomGetBody(o1);
47 dBodyID b2 = dGeomGetBody(o2);
48
49 if (b1 && b2 && dAreConnectedExcluding(b1, b2, dJointTypeContact))
50 return;
51
52
53 geomInfo* gi1 = (geomInfo*)dGeomGetData(o1);
54 geomInfo* gi2 = (geomInfo*)dGeomGetData(o2); // Check for trigger callbacks
55
56 PhysicsContext* ctx = (PhysicsContext*)data;
57
58 if (gi1 && gi1->triggerOnCollide) {
59 gi1->triggerOnCollide(ctx, o1, o2);
60 return; // Skip physical resolution
61 }
62 if (gi2 && gi2->triggerOnCollide) {
63 gi2->triggerOnCollide(ctx, o2, o1);
64 return; // Skip physical resolution
65 }
66
67 if (gi1 && !gi1->collidable) return;
68 if (gi2 && !gi2->collidable) return;
69
70 dContact contact[MAX_CONTACTS];
71 int numc = dCollide(o1, o2, MAX_CONTACTS, &contact[0].geom, sizeof(dContact));
72
73 if (numc > 0) {
74
75 SurfaceMaterial* mat1 = gi1->surface;
76 SurfaceMaterial* mat2 = gi2->surface;
77
78 dContact surfContact;
79 surfContact.surface.mu = sqrtf(mat1->friction * mat2->friction);
80 surfContact.surface.bounce = fmaxf(mat1->bounce, mat2->bounce);
81 surfContact.surface.bounce_vel = fminf(mat1->bounce_vel, mat2->bounce_vel);
82 surfContact.surface.slip1 = sqrtf(mat1->slip1 * mat2->slip1);
83 surfContact.surface.slip2 = sqrtf(mat1->slip2 * mat2->slip2);
84
85 for (int i = 0; i < numc; i++) {
86 contact[i].surface.mode = dContactBounce | dContactSlip1 | dContactSlip2 |
87 dContactSoftERP | dContactSoftCFM | dContactApprox1;
88
89
90 contact[i].surface.mu = surfContact.surface.mu;
91 contact[i].surface.bounce = surfContact.surface.bounce;
92 contact[i].surface.bounce_vel = surfContact.surface.bounce_vel;
93 contact[i].surface.soft_erp = 0.9;
94 contact[i].surface.soft_cfm = 0.001;
95 contact[i].surface.slip1 = surfContact.surface.slip1;
96 contact[i].surface.slip2 = surfContact.surface.slip2;
97
98 //contact[i].surface.slip1 = 0.001;
99 //contact[i].surface.slip2 = 0.001;
100 //contact[i].surface.bounce = 0.001;
101 //contact[i].surface.bounce_vel = 0.001;
102
103 dJointID c = dJointCreateContact(ctx->world, ctx->contactgroup, &contact[i]);
104 dJointAttach(c, b1, b2);
105 }
106 }
107}

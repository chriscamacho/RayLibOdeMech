32// custom struct to save with physics context for use in trigger callback
33typedef struct grabberData {
34 dGeomID planeGeom;
35 dJointID attachment;
36 entity* grabber;
37} grabberData;

41// this will end up joining to the first collided in a frame (could be any!)
42void triggerCallback(PhysicsContext* physCtx, dGeomID trigger, dGeomID intruder)
43{
44 (void)trigger;
45 grabberData* gdata = (grabberData*)physCtx->data;
46
47 if (intruder == gdata->planeGeom) return;
48 geomInfo* gi = dGeomGetData(intruder);
49 if (gi) {
50 gi->hew = RED;
51 if (!gdata->attachment && !IsKeyDown(KEY_G)) {
52 dBodyID bdy = dGeomGetBody(intruder);
53 gdata->attachment = dJointCreateBall (physCtx->world, 0);
54
55 const dReal* pos = dBodyGetPosition(bdy);
56
57 dJointSetBallAnchor (gdata->attachment, pos[1], pos[2], pos[3]);
58 dJointAttach(gdata->attachment, bdy, gdata->grabber->body);
59 }
60 }
61}

64int main(void)
65{
66 // Initialization
67 //--------------------------------------------------------------------------------------
68
69
70
71 GraphicsContext* graphics = CreateGraphics(screenWidth, screenHeight, "Raylib and OpenDE");
72 SetupCamera(graphics);
73 PhysicsContext* physCtx = CreatePhysics();
74
75 grabberData gData;
76 physCtx->data = &gData;
77
78 // set up for the items in the world
79 //--------------------------------------------------------------------------------------
80
81 // Create ground "plane"
82 gData.planeGeom = dCreateBox(physCtx->space, PLANE_SIZE, PLANE_THICKNESS, PLANE_SIZE);
83 dGeomSetPosition(gData.planeGeom, 0, -PLANE_THICKNESS / 2.0, 0);
84 dGeomSetData(gData.planeGeom, CreateGeomInfo(true, &graphics->groundTexture, 25.0f, 25.0f));
85
86 clistAddNode(physCtx->statics, gData.planeGeom);
87
88
89 // Create random simple objects with random textures
90 for (int i = 0; i < NUM_OBJ/4; i++) {
91 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(5, 11), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
92 }
93
94 // creation of the arm
95
96 entity* platform = CreateCylinder(physCtx, graphics, 1,//radius
97 .5,//length
98 (Vector3){0,1.5,0}, // pos
99 (Vector3){0,0,0}, // rot
100 4); // mass
101
102
103
104 dMatrix3 Rr;
105 dRFromAxisAndAngle(Rr, 1, 0, 0, M_PI/2.0);
106 dBodySetRotation (platform->body, Rr);
107
108 dJointID platform_joint = CreateRotor(physCtx, platform,0, (Vector3){0,1,0});
109
110 // platform boot is just cosmetic to hide the joint
111 dGeomID plat_boot = CreateSphereGeom(physCtx, graphics, .6, (Vector3){0,0,0});
112 dGeomSetBody(plat_boot, platform->body);
113 dGeomSetOffsetPosition(plat_boot, 0, 0, -0.5);
114 geomInfo* pbgi = dGeomGetData(plat_boot);
115 pbgi->collidable = false;
116
117
118
119 entity* rotor2 = CreateBox(physCtx, graphics,
120 (Vector3){.5, 4,.5}, // size
121 (Vector3){0,2,0}, // pos
122 (Vector3){0,0,0}, // rot
123 4); // mass
124
125 dGeomID rgeom2 = dBodyGetFirstGeom(rotor2->body);
126 dGeomSetOffsetPosition(rgeom2, 0, 2, 0);
127
128 dJointID rotor_joint2 = CreateRotor(physCtx, rotor2, platform,(Vector3){1,0,0});
129
130
131 entity* rotor3 = CreateBox(physCtx, graphics,
132 (Vector3){.5, 4,.5}, // size
133 (Vector3){0,6,0}, // pos
134 (Vector3){0,0,0}, // rot
135 4); // mass
136
137 dGeomID rgeom3 = dBodyGetFirstGeom(rotor3->body);
138 dGeomSetOffsetPosition(rgeom3, 0, 2, 0);
139
140 dJointID rotor_joint3 = CreateRotor(physCtx, rotor3, rotor2,(Vector3){1,0,0});
141
142
143 // a ball for a grabber includes a sensor to grab stuff automatically
144 gData.grabber = CreateSphere(physCtx, graphics, .5, (Vector3){0,10.5,0}, (Vector3){0,0,0}, 1);
145 dBodySetAngularDamping(gData.grabber->body, 0.5f); // make it stiffer
146
147 dJointID grabber_joint = dJointCreateBall (physCtx->world, 0);
148 dJointSetBallAnchor (grabber_joint, 0, 10.5, 0);
149 dJointAttach(grabber_joint, gData.grabber->body, rotor3->body);
150
151
152 // create just a geom
153 dGeomID grab_sensor = CreateSphereGeom(physCtx, graphics, .6, (Vector3){0,0,0});
154
155 // set the trigger callback for the geom its now invisible
156 // and other things don't collide with it - but it will register "collisions"
157 geomInfo* triggergi = dGeomGetData(grab_sensor);
158 triggergi->triggerOnCollide = triggerCallback;
159 dGeomSetBody(grab_sensor, gData.grabber->body); // attach the sensor to the grabber
160
161 float physTime = 0;
162
163 //--------------------------------------------------------------------------------------
164 //
165 // Main game loop
166 //
167 //--------------------------------------------------------------------------------------
168 while (!WindowShouldClose()) // Detect window close button or ESC key
169 {
170 //--------------------------------------------------------------------------------------
171 // Update
172 //----------------------------------------------------------------------------------
173
174 // baked in controls (example camera)
175 UpdateCameraControl(graphics);
176
177 if (IsKeyDown(KEY_G)) {
178 if (gData.attachment) {
179 dJointDestroy(gData.attachment);
180 gData.attachment = 0;
181 }
182 }
183
184 dJointSetHingeParam(platform_joint, dParamVel, 0.0);
185 dJointSetHingeParam(rotor_joint2, dParamVel, 0.0);
186 dJointSetHingeParam(rotor_joint3, dParamVel, 0.0);
187
188 if (IsKeyDown(KEY_I)) {
189 dJointSetHingeParam(platform_joint, dParamVel, 1.0);
190 dBodyEnable(platform->body);
191 }
192
193 if (IsKeyDown(KEY_O)) {
194 dJointSetHingeParam(platform_joint, dParamVel, -1.0);
195 dBodyEnable(platform->body);
196 }
197
198 if (IsKeyDown(KEY_K)) {
199 dJointSetHingeParam(rotor_joint2, dParamVel, 1.0);
200 dBodyEnable(rotor2->body);
201 }
202
203 if (IsKeyDown(KEY_L)) {
204 dJointSetHingeParam(rotor_joint2, dParamVel, -1.0);
205 dBodyEnable(rotor2->body);
206 }
207
208 if (IsKeyDown(KEY_COMMA)) {
209 dJointSetHingeParam(rotor_joint3, dParamVel, 1.0);
210 dBodyEnable(rotor3->body);
211 }
212
213 if (IsKeyDown(KEY_PERIOD)) {
214 dJointSetHingeParam(rotor_joint3, dParamVel, -1.0);
215 dBodyEnable(rotor3->body);
216 }
217
218 bool spcdn = IsKeyDown(KEY_SPACE); // cache space key status (don't look up for each object iterration
219 cnode_t* node = physCtx->objList->head;
220
221 while (node != NULL) {
222 entity* ent = node->data;
223 dBodyID bdy = ent->body;
224 cnode_t* next = node->next; // get the next node now in case we delete this one
225
226 // reset entities for setting by collision
227 SetEntityHew(ent, WHITE);
228
229 const dReal* pos = dBodyGetPosition(bdy);
230 if (spcdn) {
231 // apply force if the space key is held
232 const dReal* v = dBodyGetLinearVel(bdy);
233 if (v[1] < 10 && pos[1]<10) { // cap upwards velocity and don't let it get too high
234 dBodyEnable (bdy); // case its gone to sleep
235 dMass mass;
236 dBodyGetMass (bdy, &mass);
237 // give some object more force than others
238 float f = rndf(8,20) * mass.mass;
239 dBodyAddForce(bdy, rndf(-f,f), f*10, rndf(-f,f));
240 }
241 }
242
243
244 if(pos[1]<-10) {
245 FreeEntity(physCtx, ent); // warning deletes global entity list entry, get your next node before doing this!
246 CreateRandomEntity(physCtx, graphics, (Vector3){rndf(-3, 3), rndf(6, 12), rndf(-3, 3)}, SHAPE_ALL);
247 }
248
249 node = next;
250 }
251
252 // Step the physics
253 //----------------------------------------------------------------------------------
254
255 physTime = GetTime();
256 int pSteps = StepPhysics(physCtx);
257 physTime = GetTime() - physTime;
258
259
260 // Draw
261 //----------------------------------------------------------------------------------
262
263 BeginDrawing();
264
265 ClearBackground(BLACK);
266
267 BeginMode3D(graphics->camera);
268 DrawBodies(graphics, physCtx);
269 DrawStatics(graphics, physCtx);
270 EndMode3D();
271
272
273 if (pSteps > maxPsteps) DrawText("WARNING CPU overloaded lagging real time", 10, 0, 20, RED);
274 DrawText(TextFormat("%2i FPS", GetFPS()), 10, 20, 20, WHITE);
275 DrawText("Press SPACE to apply force to objects", 10, 60, 20, WHITE);
276
277 DrawText(TextFormat("Phys steps per frame %i",pSteps), 10, 120, 20, WHITE);
278 DrawText(TextFormat("Phys time per frame %f",physTime), 10, 140, 20, WHITE);
279 DrawText(TextFormat("total time per frame %f",GetFrameTime()), 10, 160, 20, WHITE);
280
281 EndDrawing();
282
283 }
284
285
286 // De-Initialization
287 //--------------------------------------------------------------------------------------
288 FreePhysics(physCtx);
289 FreeGraphics(graphics);
290
291 CloseWindow(); // Close window and OpenGL context
292
293 return 0;
294}

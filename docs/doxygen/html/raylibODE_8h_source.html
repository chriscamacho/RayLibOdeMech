78/**
79 * @brief Function pointer type for physics trigger events
80 * * This callback is triggered when a geometry designated as a "ghost" or sensor
81 * overlaps with another geometry. When this callback is set in geomInfo,
82 * the physics engine will skip standard contact resolution, allowing
83 * objects to pass through while still notifying game logic.
84 * @param trigger The dGeomID of the geometry that has the callback assigned.
85 * @param intruder The dGeomID of the geometry that entered or is inside the trigger.
86 * @note If the triggering geom is attached to a body, the trigger will follow the body's transform.
87 * @note The physics context is passed with the callback, which has a user data pointer
88 * this can be used to avoid having to use global variables in the callback
89 * @see geomInfo
90 */
91typedef void (*TriggerCallback)(PhysicsContext* pctx, dGeomID trigger, dGeomID intruder);

93/**
94 * @brief Stores geometry metadata for collision, texturing, and trimesh data.
95 * * This structure encapsulates everything needed to render and handle physics
96 * for a specific piece of geometry, including UV scaling and raw mesh indices.
97 */
98typedef struct geomInfo {
99 bool collidable; /**< Toggle for physics engine interaction. */
100 Texture* texture; /**< Pointer to the associated diffuse/albedo map. set to null for invisible geom*/
101 float uvScaleU; /**< Horizontal texture tiling factor. */
102 float uvScaleV; /**< Vertical texture tiling factor. */
103 Model visual; /**< Model override used for custom static trimeshes. */
104 Color hew; /**< white for normal, used to tint a geom */
105 SurfaceMaterial* surface; /**< friction, restitution and so on */
106
107 /** @name Trimesh Encapsulation
108 * Members used specifically for raw triangle mesh data.
109 * @{ */
110 int* indices; /**< Pointer to the array of vertex indices. */
111 dTriMeshDataID triData; /**< ODE/Physics-specific trimesh data identifier. */
112 /** @} */
113
114 TriggerCallback triggerOnCollide; /**< If this is non-NULL, the geom acts as a ghost/trigger */
115 void* data; /**< user data pointer tag on extra meta data to a geom. */
116} geomInfo;

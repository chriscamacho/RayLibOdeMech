49/**
50 * @brief Function pointer type for physics trigger events
51 * * This callback is triggered when a geometry designated as a "ghost" or sensor
52 * overlaps with another geometry. When this callback is set in geomInfo,
53 * the physics engine will skip standard contact resolution, allowing
54 * objects to pass through while still notifying game logic.
55 * * @param trigger The dGeomID of the geometry that has the callback assigned.
56 * @param intruder The dGeomID of the geometry that entered or is inside the trigger.
57 * * @note If attached to a dBodyID, the trigger will follow the body's transform.
58 * @see geomInfo
59 */
60typedef void (*TriggerCallback)(dGeomID trigger, dGeomID intruder);

62/**
63 * @brief Stores geometry metadata for collision, texturing, and trimesh data.
64 * * This structure encapsulates everything needed to render and handle physics
65 * for a specific piece of geometry, including UV scaling and raw mesh indices.
66 */
67typedef struct geomInfo {
68 bool collidable; /**< Toggle for physics engine interaction. */
69 Texture* texture; /**< Pointer to the associated diffuse/albedo map. set to null for invisible geom*/
70 float uvScaleU; /**< Horizontal texture tiling factor. */
71 float uvScaleV; /**< Vertical texture tiling factor. */
72 Model visual; /**< Model override used for custom static trimeshes. */
73 Color hew; /**< white for normal, used to tint a geom */
74 /** @name Trimesh Encapsulation
75 * Members used specifically for raw triangle mesh data.
76 * @{ */
77 int* indices; /**< Pointer to the array of vertex indices. */
78 dTriMeshDataID triData; /**< ODE/Physics-specific trimesh data identifier. */
79 /** @} */
80
81 TriggerCallback triggerOnCollide; /**< If this is non-NULL, the geom acts as a ghost/trigger */
82 void* data; /**< user data pointer tag on extra meta data to a geom. */
83} geomInfo;

52/**
53 * @brief Function pointer type for physics trigger events
54 * * This callback is triggered when a geometry designated as a "ghost" or sensor
55 * overlaps with another geometry. When this callback is set in geomInfo,
56 * the physics engine will skip standard contact resolution, allowing
57 * objects to pass through while still notifying game logic.
58 * * @param trigger The dGeomID of the geometry that has the callback assigned.
59 * @param intruder The dGeomID of the geometry that entered or is inside the trigger.
60 * * @note If attached to a dBodyID, the trigger will follow the body's transform.
61 * @see geomInfo
62 */
63typedef void (*TriggerCallback)(dGeomID trigger, dGeomID intruder);

65/**
66 * @brief Stores geometry metadata for collision, texturing, and trimesh data.
67 * * This structure encapsulates everything needed to render and handle physics
68 * for a specific piece of geometry, including UV scaling and raw mesh indices.
69 */
70typedef struct geomInfo {
71 bool collidable; /**< Toggle for physics engine interaction. */
72 Texture* texture; /**< Pointer to the associated diffuse/albedo map. set to null for invisible geom*/
73 float uvScaleU; /**< Horizontal texture tiling factor. */
74 float uvScaleV; /**< Vertical texture tiling factor. */
75 Model visual; /**< Model override used for custom static trimeshes. */
76 Color hew; /**< white for normal, used to tint a geom */
77 /** @name Trimesh Encapsulation
78 * Members used specifically for raw triangle mesh data.
79 * @{ */
80 int* indices; /**< Pointer to the array of vertex indices. */
81 dTriMeshDataID triData; /**< ODE/Physics-specific trimesh data identifier. */
82 /** @} */
83
84 TriggerCallback triggerOnCollide; /**< If this is non-NULL, the geom acts as a ghost/trigger */
85 void* data; /**< user data pointer tag on extra meta data to a geom. */
86} geomInfo;

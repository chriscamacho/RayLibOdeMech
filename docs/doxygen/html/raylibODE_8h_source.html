63/**
64 * @brief Function pointer type for physics trigger events
65 * * This callback is triggered when a geometry designated as a "ghost" or sensor
66 * overlaps with another geometry. When this callback is set in geomInfo,
67 * the physics engine will skip standard contact resolution, allowing
68 * objects to pass through while still notifying game logic.
69 * @param trigger The dGeomID of the geometry that has the callback assigned.
70 * @param intruder The dGeomID of the geometry that entered or is inside the trigger.
71 * @note If the triggering geom is attached to a body, the trigger will follow the body's transform.
72 * @note The physics context is passed with the callback, which has a user data pointer
73 * this can be used to avoid having to use global variables in the callback
74 * @see geomInfo
75 */
76typedef void (*TriggerCallback)(PhysicsContext* pctx, dGeomID trigger, dGeomID intruder);

78/**
79 * @brief Stores geometry metadata for collision, texturing, and trimesh data.
80 * * This structure encapsulates everything needed to render and handle physics
81 * for a specific piece of geometry, including UV scaling and raw mesh indices.
82 */
83typedef struct geomInfo {
84 bool collidable; /**< Toggle for physics engine interaction. */
85 Texture* texture; /**< Pointer to the associated diffuse/albedo map. set to null for invisible geom*/
86 float uvScaleU; /**< Horizontal texture tiling factor. */
87 float uvScaleV; /**< Vertical texture tiling factor. */
88 Model visual; /**< Model override used for custom static trimeshes. */
89 Color hew; /**< white for normal, used to tint a geom */
90 /** @name Trimesh Encapsulation
91 * Members used specifically for raw triangle mesh data.
92 * @{ */
93 int* indices; /**< Pointer to the array of vertex indices. */
94 dTriMeshDataID triData; /**< ODE/Physics-specific trimesh data identifier. */
95 /** @} */
96
97 TriggerCallback triggerOnCollide; /**< If this is non-NULL, the geom acts as a ghost/trigger */
98 void* data; /**< user data pointer tag on extra meta data to a geom. */
99} geomInfo;
